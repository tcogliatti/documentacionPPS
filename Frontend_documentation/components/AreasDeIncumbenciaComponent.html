import { Component, EventEmitter, OnInit, Output, ViewChild } from '@angular/core';
import { Router } from '@angular/router';
import { Area } from 'src/app/interfaces/Area';
import { ROUTES } from 'src/constants';
import { AreasService } from '../../../servicios/areas.service';
import { PostulanteService } from 'src/app/servicios/postulante.service';
import { FormAreasDeIncumbenciaComponent } from '../../form-areas-de-incumbencia/form-areas-de-incumbencia.component';

@Component({
  selector: 'app-areas-de-incumbencia',
  templateUrl: './areas-de-incumbencia.component.html',
  styleUrls: ['./areas-de-incumbencia.component.css']
})

/**
 * Se encarga de contener la carga de areas de incumbencia
 * Se encarga de renderizar los mensajes de Loading y error 404
 * Se encarga de realizar el fetching de datos a la base de datos para renderizar el formulario con los datos de las opciones
 */
export class AreasDeIncumbenciaComponent implements OnInit {
  /**
   * Varibale local que apunta a la constante ROUTES en constants.ts
   */
  routes: any = ROUTES;

  /**
   * Areas que se obtendran del backend para que el usuario las seleccione
   */
  areas: Area[] = [];

  /**
   * URL para el boton "Volver"
   */
  urlAnterior: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.DATOS_CONTACTO}`;

  /**
   * URL para el boton "Recargar" de la pantalla de error
   */
  urlActual: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.AREAS_INCUMBENCIA}`;

  /**
   * URL para el boton finalizar, el cual renderizara el componente pantalla de fin de carga
   */
  urlFinalizar: string = `/${ROUTES.COMPLETO_DATOS_REQUERIDOS}`;

  /**
   * Flag que renderizara condicionalmente el componente de error
   */
  httpError: boolean = false;

  /**
   * Flag que renderizara condicionalmente el componente de Loader
   */
  isLoading: boolean = true;

  /**
   * Instacia del SelectAtucompleteComponent
   */
  @ViewChild(FormAreasDeIncumbenciaComponent) childComponent!: FormAreasDeIncumbenciaComponent;

  @Output()
  emitUrlReload: EventEmitter<string> = new EventEmitter<string>();

  /**
   * Creamos el componente inyectando AreaService, Router, PostulanteService
   */
  constructor(private AreasService: AreasService, private router: Router, private postulanteService: PostulanteService) {
  }

  async ngOnInit(): Promise<void> {

    try {
      this.isLoading = true;
      this.areas = await this.AreasService.getAreas();
      window.scrollTo(0, 0);
      this.httpError = false;
      this.isLoading = false;
    } catch (error) {
      console.error('Error: ', error);
      this.httpError = true;
      this.isLoading = false;
      this.emitUrlReload.emit(this.urlActual)
    }
  }
  /**
* Esta funcion se encargara de guardar (haciendo uso del service del postulante),
* el atributo "this.areasElegidas".
*/
  async enviar() {
    if (this.childComponent.actualizarInfo()) {
      // postulante Service Post new Postulante
      try {
        this.isLoading = true;
        await this.postulanteService.persistPostulante();
        this.isLoading = false;
        this.router.navigate([this.urlFinalizar]);
      } catch (error) {
        this.isLoading = false;
        console.error('Error: ', error);
        this.httpError = true;
        this.emitUrlReload.emit(this.urlActual)
      }
    }
  }

  /**
   * Actualiza el observable y avisa al postulanteService del cambio
   */
  updatePostualanteLocal() {
    this.childComponent.updatePostualanteLocal();
  }

  /**
   * Redireccion dada pro el boton "Volver"
   */
  volver(): void {
    this.updatePostualanteLocal()
    this.router.navigate([this.urlAnterior]);
  }
}
