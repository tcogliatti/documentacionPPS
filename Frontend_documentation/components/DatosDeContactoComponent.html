import { Component, EventEmitter, OnInit, Output, QueryList, ViewChild, ViewChildren } from '@angular/core';
import { Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
import { Calle } from 'src/app/interfaces/Calle';
import { Localidad } from 'src/app/interfaces/Localidad';
import { Partido } from 'src/app/interfaces/Partido';
import { PostulantePerfil } from 'src/app/interfaces/PostulantePerfil';
import { Provincia } from 'src/app/interfaces/Provincia';
import { OptionFormService } from 'src/app/servicios/option-form.service';
import { PostulanteService } from 'src/app/servicios/postulante.service';
import { ValidarFormHelperService } from 'src/app/servicios/validar-form-helper.service';
import { MINIMOS, ROUTES } from 'src/constants';
import { SelectAutocompleteComponent } from '../../select-autocomplete/select-autocomplete.component';
import { FormDatosDeContactoComponent } from "../../form-datos-de-contacto/form-datos-de-contacto.component";

@Component({
  selector: 'app-datos-de-contacto',
  templateUrl: './datos-de-contacto.component.html',
  styleUrls: ['./datos-de-contacto.component.css']
})

/**
 * Se encarga de contener el formulario para la carga de los datos de contacto
 * Se encarga de renderizar los mensajes de Loading y error 404
 * Se encarga de realizar el fetching de datos a la base de datos para renderizar el formulario con los datos de las opciones
 */
export class DatosDeContactoComponent implements OnInit {

  // url para dirigirse al formulario siguiente de áreas de incumbencias
  urlFormAreas: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.AREAS_INCUMBENCIA}`;
  // url para dirigirse al formulario anterior de datos personales
  urlFormDatosPersonales: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.DATOS_PERSONALES}`;

  httpError: boolean = false;
  isLoading: boolean = true;

  urlReload: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.DATOS_CONTACTO}`;

  provincias: Provincia[] = [];
  localidades: Localidad[] = [];
  partidos: Partido[] = [];
  calles: Calle[] = [];

  @Output()
  emitUrlReload: EventEmitter<string> = new EventEmitter<string>();

  routes: any = ROUTES;

  enviado = false; //lo necesita el required del componente ng-autocomplete

  @ViewChild(FormDatosDeContactoComponent) childComponent!: FormDatosDeContactoComponent;


  constructor(private optFormServ: OptionFormService, private router: Router) {
    this.isLoading = false;
    this.httpError = false;
  }

  async ngOnInit(): Promise<void> {
    /**
     * Obtener los datos desde el backend
     */
    try {
      this.isLoading = true;

      this.provincias = await this.optFormServ.getProvincias()
      this.partidos = await this.optFormServ.getPartidos()
      this.localidades = await this.optFormServ.getLocalidades()
      this.calles = await this.optFormServ.getCalles()

      window.scrollTo(0, 0);

      this.httpError = false;
      this.isLoading = false;

    } catch (error) {
      console.error('Error: ', error);
      this.httpError = true;
      this.isLoading = false;
      //this.emitUrlReload.emit(this.urlReload)
    }
  }

  enviar() {
    let valido: boolean = this.childComponent.actualizarInfo()
    if (valido) {
      // redirige a formulario de áreas de incumbencia
      this.router.navigate([this.urlFormAreas]);
    }
  }

  volver(): void {

    // redirige a formulario de áreas de incumbencia (Sin guardar los cambios)
    this.router.navigate([this.urlFormDatosPersonales]);
  }

}
