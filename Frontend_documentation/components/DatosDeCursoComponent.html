import { Component, OnInit, ViewChild } from '@angular/core';
import { Router } from '@angular/router';
import { ROUTES } from 'src/constants';
import { Curso } from 'src/app/interfaces/Curso';
import { FormDatosCvCursoComponent } from '../../form-datos-cv-curso/form-datos-cv-curso.component';

@Component({
  selector: 'app-datos-de-curso',
  templateUrl: './datos-de-curso.component.html',
  styleUrls: ['./datos-de-curso.component.css']
})


/**
 * Se encarga de contener el formulario y las cargas de los cursos
 * Se encarga de renderizar los mensajes de Loading y error 404
 * Se encarga de realizar el fetching de datos a la base de datos para renderizar el formulario con los datos de las opciones
 */
export class DatosDeCursoComponent implements OnInit {
 
  isLoading: boolean = false;
  httpError: boolean = false;
  // url datos cv curso
  urlReload: string = `/${ROUTES.DATOS_CV}/${ROUTES.CURSO}`;

  // url para dirigirse al formulario siguiente de experiencia laboral
  urlFormExperienciaLaboral: string = `/${ROUTES.DATOS_CV}/${ROUTES.EXPERIENCIA_LABORAL}`;
  // url para dirigirse al formulario anterior de nivel educativo
  urlFormNivelEducativo: string = `/${ROUTES.DATOS_CV}/${ROUTES.NIVEL_EDUCATIVO}`;

  /* id inicial del curso actual */
  id: number = 0;
  /* variable para guardar el id del ultimo curso agregado */
  ultimoID: number = 0;
  /* arreglo de anios para mostrar en el select */
  anios: number[] = [];

  cursoAeliminar: Curso | undefined;

  /* flag que cambia el titulo entre "agregar curso" y "editar curso" */
  editandoFlag: boolean = false;

  cursos: Curso[] = [];

  /**
   * este es el objeto del curso actual, el cual se actualiza por medio de :
     two way data binding al actualizar los valores de los inputs 
   */
  cursoActual: Curso = {
    idcurso: 0,
    idpostulante: 0,
    descripcion: "",
    establecimiento: "",
    duracion: "",
    anio: 0,
    estado: "",
  }

  constructor(private router: Router) { 
  
    this.httpError = false;
    this.isLoading = false;
  }

  ngOnInit(): void {
    window.scrollTo(0,0);
    try{
      this.isLoading = false;
      this.httpError = false;
    }catch(error){
      this.isLoading = true;
      this.httpError = true;  
      console.log("Error: " + error);
    }

  }

  /**
   * acceso al hijo del tipo : FormDatosCvNivelEducativoComponent
   */
  @ViewChild(FormDatosCvCursoComponent) childComponent!: FormDatosCvCursoComponent;
  
  
  /**
   *  boton que redirige hacia la seccion de experiencia laboral 
   */
  enviar() {
    if(this.childComponent.actualizarInfo()){      
      try {
        this.router.navigate([this.urlFormExperienciaLaboral]);
      } catch (error) {
          this.httpError = true;
      }
    } 
  }

  /**
   * boton que redirige la seccion de nivel educativo
   */
  volver() {
    this.router.navigate([this.urlFormNivelEducativo]);
  }
}

        <app-form-datos-cv-curso></app-form-datos-cv-curso>
