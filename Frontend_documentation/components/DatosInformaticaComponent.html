import { Component, OnInit, Output, ViewChild } from '@angular/core';
import { ROUTES } from '../../../../constants'
import { Router } from '@angular/router';
import {  } from '../../form-experiencia-laboral/form-experiencia-laboral.component';
import { FormInformaticaComponent } from '../../form-informatica/form-informatica.component';

@Component({
  selector: 'app-datos-informatica',
  templateUrl: './datos-informatica.component.html',
  styleUrls: ['./datos-informatica.component.css']
})

/**
 * Se encarga de contener el formulario y las cargas de los conocimientos unformáticos
 * Se encarga de renderizar los mensajes de Loading y error 404
 * Se encarga de realizar el fetching de datos a la base de datos para renderizar el formulario con los datos de las opciones
 */
export class DatosInformaticaComponent implements OnInit {
  /**
   * URLs para redireccionar la navegación
   */
  urlFormAnterior: string = `/${ROUTES.DATOS_CV}/${ROUTES.IDIOMA}`
  urlFormActual: string = `${ROUTES.DATOS_CV}/${ROUTES.INFORMATICA}`
  urlFormSiguiente: string = `/${ROUTES.COMPLETO_DATOS_CV}`

  
  /**
   * Flag que renderizara condicionalmente el componente de error
   */
  httpError: boolean = false;

  /**
   * Flag que renderizara condicionalmente el componente de Loader
   */
  isLoading: boolean = false;

  /**
  * acceso al hijo del tipo : FormExperienciaLaboralComponent
  */
 @ViewChild(FormInformaticaComponent) childComponent!: FormInformaticaComponent;


  constructor(private router: Router) { }

  async ngOnInit(): Promise<void> { 
    window.scrollTo(0,0)
  }

  /**
   * TODO: Cuando el componente se separe, esta debera actualizar la info 
   */
  enviar() {
    if (this.childComponent.posInfo()) {
      try {
        this.router.navigate([this.urlFormSiguiente]);
      } catch (error) {
        this.httpError = true;
      }
    }
  }

  volver() {
    if (this.childComponent.actualizarInfo()) {
      try {
        this.router.navigate([this.urlFormAnterior]);
      } catch (error) {
        this.httpError = true;
      }
    }
  }
}
