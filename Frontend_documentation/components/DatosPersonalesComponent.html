import { Component, EventEmitter, OnInit, Output, QueryList, ViewChild, ViewChildren } from '@angular/core';
import { ROUTES } from 'src/constants';
import { Router } from '@angular/router';
import { FormDatosPersonalesComponent } from 'src/app/formularios/form-datos-personales/form-datos-personales.component';
import { Pais } from 'src/app/interfaces/Pais';
import { TipoDocumento } from 'src/app/interfaces/TipoDocumento';
import { EstadoCivil } from 'src/app/interfaces/EstadoCivil';
import { Genero } from 'src/app/interfaces/Genero';
import { OptionFormService } from 'src/app/servicios/option-form.service';

@Component({
  selector: 'app-datos-personales',
  templateUrl: './datos-personales.component.html',
  styleUrls: ['./datos-personales.component.css']
})

/**
 * Se encarga de contener el formulario para la carga de los datos personales
 * Se encarga de renderizar los mensajes de Loading y error 404
 * Se encarga de realizar el fetching de datos a la base de datos para renderizar el formulario con los datos de las opciones
 */
export class DatosPersonalesComponent implements OnInit {

  // url para dirigirse al formulario de contacto
  urlFormContact: string = "/" + ROUTES.DATOS_REQUERIDOS + "/" + ROUTES.DATOS_CONTACTO;

  isLoading: boolean = true;
  httpError: boolean = false;

  // url datos personales
  urlReload: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.DATOS_PERSONALES}`

  paises: Pais[] = [];
  tiposDoc: TipoDocumento[] = [];
  estadosCiviles: EstadoCivil[] = [];
  generos: Genero[] = [];

  @Output()
  emitUrlReload: EventEmitter<string> = new EventEmitter<string>();

  @ViewChild(FormDatosPersonalesComponent) childComponent!: FormDatosPersonalesComponent;

  constructor(private router: Router, private optFormServ: OptionFormService) {
    this.isLoading = false;
    this.httpError = false;
  }
  
  /**
   * Obtiene datos cargados en local storage si es que existen
   */
  async ngOnInit(): Promise<void> {
    try {
      this.isLoading = true;

      this.tiposDoc = await this.optFormServ.getTipoDoc();
      this.estadosCiviles = await this.optFormServ.getEstadoCivil();
      this.paises = await this.optFormServ.getPaises();
      this.generos = await this.optFormServ.getGeneros();

      window.scrollTo(0, 0);

      this.isLoading = false;
      this.httpError = false;

    } catch (error) {
      console.error('Error: ', error);
      this.httpError = true;
      this.isLoading = false;
    }
  }

  /* ---------------------------------------------- ENVIO Y VALIDACION ----------------------------------------------------  */

  /**
   * Metodo llamado al presionar en el boton de siguiente
  */
  enviar() {
    let valido = this.childComponent.actualizarInfo();
    if (valido) {
      // Redidige a formulario de datos de contacto
      this.router.navigate([this.urlFormContact]);
    }
  }

}
