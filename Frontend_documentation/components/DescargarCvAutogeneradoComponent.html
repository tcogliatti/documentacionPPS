import { Component, OnInit } from '@angular/core';
import { ROUTES } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
import { PostulantePerfil } from 'src/app/interfaces/PostulantePerfil';
import { PostulanteService } from 'src/app/servicios/postulante.service';
import { API_BASE } from 'src/constants';
import pdfMake from 'pdfmake/build/pdfmake';
import pdfFonts from 'pdfmake/build/vfs_fonts';
import htmlToPdfmake from 'html-to-pdfmake'
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { ConocimientoInformatica } from 'src/app/interfaces/ConocimientoInformatica';
import { ExperienciaLaboral } from 'src/app/interfaces/ExperienciaLaboral';
import { Curso } from 'src/app/interfaces/Curso';
import { EducacionPostulante } from 'src/app/interfaces/EducacionPostulante';
import { IdiomaPostulante } from 'src/app/interfaces/IdiomaPostulante';
pdfMake.vfs = pdfFonts.pdfMake.vfs;

@Component({
  selector: 'app-descargar-cv-autogenerado',
  templateUrl: './descargar-cv-autogenerado.component.html',
  styleUrls: ['./descargar-cv-autogenerado.component.css']
})
export class DescargarCvAutogeneradoComponent implements OnInit {

  httpError: boolean = false;
  isLoading: boolean = false;

  routes: any = ROUTES;
  previewImage: string | null = null;

  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteForm!: PostulantePerfil;

  datosAdicionales: boolean = false;
  existPostulante$: Observable<boolean>;

  nombreArchivo!: string;

  /**
   * Datos CV Servicio
   */
  datosCVObservable$!: Observable<DatosAdicionales>;
  datosCVSuscription!: Subscription;

  // colecciones de datos adicionales
  conocimientosEnInformatica: ConocimientoInformatica[] = [];
  experienciaLaboral: ExperienciaLaboral[] = [];
  cursos: Curso[] = []
  nivelEduicativo: EducacionPostulante[] = []
  idiomas: IdiomaPostulante[] = []

  constructor(private postulanteService: PostulanteService, private datosCVService: DatosAdicionalesService) {

    // obtener datos requeridos
    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable()
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(postulanteFromService => {
      this.postulanteForm = { ...postulanteFromService }

      this.nombreArchivo = `CV-${this.postulanteForm.apellido}-${this.postulanteForm.nombres}-autogenerado.pdf`

      // verifica que la foto de perfil exista, y le aplica la ruta completa
      if (this.postulanteForm.foto.path || this.postulanteForm.foto.path != "")
        this.previewImage = API_BASE + this.postulanteForm.foto.path;
      else
        this.previewImage = null
    })

    this.existPostulante$ = this.postulanteService.tienePerfil.asObservable();

    // obtener datos CV
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      console.log(datosCVFormService);
      this.conocimientosEnInformatica = datosCVFormService.conocimientosInformatica;
      this.nivelEduicativo = datosCVFormService.educacionPostulante;
      this.cursos = datosCVFormService.curso;
      this.experienciaLaboral = datosCVFormService.experienciaLaboral;
      this.idiomas = datosCVFormService.idioma;
    });
  }

  async ngOnInit(): Promise<void> {
    this.datosCVService.obtenerDatosCVByIdPostulante();
    window.scroll(0, 0);
  }

  public openCvPdf(): void {
    const cvPDF = this.generateCvPdf();
    cvPDF.open()
  }

  private generateCvPdf(): any {
    const data = document.querySelector('#htmlToPdf').outerHTML

    const html = htmlToPdfmake(
      `${data}`,
      {
        imagesByReference: true
      }
    );

    const configPdf: any = {
      info: {
        title: this.nombreArchivo,
        author: 'Sistema Empleo - Municipalidad de Olavarría',
      },
      content: html.content,
      images: html.images,
      styles: {
        "foto-user": {
          alignment: "center"
        },
        "texto-oscuro": {
          alignment: "right"
        },
        "nombre-apellido": {
          alignment: "center",
          margin: [0, 30, 0, 30]
        },
        "tabla-header": {
          with: "*",
          alignment: "center",
          justify: "center"
        }
      }
    }
    const cvPDF = pdfMake.createPdf(configPdf)
    return cvPDF
  }

  public downloadCvPdf(): any {
    const cvPDF = this.generateCvPdf();
    cvPDF.download(this.nombreArchivo)
  }

}
