import { Component, OnInit, ViewChild } from '@angular/core';
import { Router, RouterOutlet } from '@angular/router';
import { FormDatosPersonalesComponent } from '../formularios/form-datos-personales/form-datos-personales.component';
import { OptionFormService } from '../servicios/option-form.service';
import { ROUTES } from 'src/constants';
import { PostulanteService } from '../servicios/postulante.service';
import {
  FormAreasDeIncumbenciaComponent
} from '../formularios/form-areas-de-incumbencia/form-areas-de-incumbencia.component';
import { AreasService } from '../servicios/areas.service';
import { FormDatosDeContactoComponent } from "../formularios/form-datos-de-contacto/form-datos-de-contacto.component";
import { FormDatosCvNivelEducativoComponent } from '../formularios/form-datos-cv-nivel-educativo/form-datos-cv-nivel-educativo.component';
import { FormExperienciaLaboralComponent } from '../formularios/form-experiencia-laboral/form-experiencia-laboral.component';
import { FormInformaticaComponent } from '../formularios/form-informatica/form-informatica.component';
import { DatosAdicionalesService } from '../servicios/datos-adicionales.service';
import { Titulos } from '../interfaces/Titulos';
import { FormDatosCvCursoComponent } from '../formularios/form-datos-cv-curso/form-datos-cv-curso.component';
import { FormDatosCvIdiomaComponent } from '../formularios/form-datos-cv-idioma/form-datos-cv-idioma.component';

/**
 * Contiene los formularios utilizados para editar los datos
 * Este componente tiene la responsabilidad de mostrar los loadings y los mensajes de error 404.
 * Para que esto ocurra este compnente debe ser el responsable de hacer el fetch a la base de datos para traer la información necesaria para renderizar las opciones de los forms y a su vez traer lso datos ya cargados por el usuario par ser editados.
 * Por esta razon el componente debe saber que cual es el formulario a editar.
 */
@Component({
  selector: 'app-editar-form',
  templateUrl: './editar-form.component.html',
  styleUrls: ['./editar-form.component.css']
})
export class EditarFormComponent implements OnInit {

  /**
   * Captura la instancia del RouterOutlet que representa la salida de las rutas hijas del componente.
   */
  @ViewChild(RouterOutlet, { static: true }) outlet!: RouterOutlet;

  /**
   * Variable accesible desde el html que reoresenta la constante ROUTES 
   */
  routes: any = ROUTES;

  /**
   * Bandera de control que renderizara condicionalemente el componente Loader
   */
  isLoading: boolean = true;

  /**
   * Bandera de control que renderizara condicionalmente el componente httpError si hubiera una falla en el back
   */
  httpError: boolean = false;

  /**
   * Ruta que se emitira al componente de httpError para recargar la pagina
   * segun el formulario que corresponda
   */
  urlReload: string = "";

  /**
   * Titulo que se renderizara en el html
   */
  editTittle: string = "";

  /**
   * Categoria actual seteada en loadDataForCurrentRoute()
   */
  actualCategory = "";

  /**
   * Estados de las categorias del formulario a actualizar
   */
  categorys = {
    REQUERIDO: "REQUERIDO",
    ADICIONAL: "ADICIONAL"
  }

  /**
   * Titulos de cada seccion que se renderizaran en el html
   */
  titles: Titulos = {
    datosPersonales: "datos personales",
    datosContacto: "datos de contacto",
    datosAreas: "areas de incumbencia",
    datosEducacion: "nivel educativo",
    datosExperiencia: "experiencia laboral",
    datosIdioma: "idioma",
    datosInformatica: "conocimientos informáticos",
    datosCurso: "curso"
  }

  actualComponentInstance: any;

  constructor(private areasService: AreasService, private optService: OptionFormService, private router: Router, private perfilPostulanteService: PostulanteService, private datosCVService: DatosAdicionalesService) {
  }

  async ngOnInit(): Promise<void> {
    try {
      //Se encarga de cargar los datos específicos necesarios para el componente que se
      // está mostrando en el RouterOutlet
      await this.loadDataForCurrentRoute();
    } catch (error) {
      console.log("Sucedió un error:", error)
      //Utilizado como bandera para mostrar errores o cargas en html
      this.isLoading = false;
      this.httpError = true;
    }
  }

  /**
   * Rutea el componenete segun corresponda corresponda y setea las opciones de menu input 
   * necesarias para cada uno de los formularios si es que lo necesita.
   */
  private async loadDataForCurrentRoute(): Promise<void> {

    // Verifica cual es el componente que se instancia en el outlet
    const componentInstance = this.outlet.component;

    try {

      // Según el componente instanciado, se configuran propiedades y se cargan datos específicos
      if (componentInstance instanceof FormDatosPersonalesComponent) {
        // FORM DATOS PERSONALES
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.DATOS_PERSONALES;
        this.editTittle = this.titles.datosPersonales;
        this.actualCategory = this.categorys.REQUERIDO;
        componentInstance.generos = await this.optService.getGeneros();
        componentInstance.tiposDoc = await this.optService.getTipoDoc();
        componentInstance.estadosCiviles = await this.optService.getEstadoCivil();
        componentInstance.paises = await this.optService.getPaises();
        this.isLoading = false;
      } else if (componentInstance instanceof FormDatosDeContactoComponent) {
        // FORM DATOS DE CONTACTO
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.DATOS_CONTACTO;
        this.editTittle = this.titles.datosContacto;
        this.actualCategory = this.categorys.REQUERIDO;
        componentInstance.calles = await this.optService.getCalles();
        componentInstance.localidades = await this.optService.getLocalidades();
        componentInstance.partidos = await this.optService.getPartidos();
        componentInstance.provincias = await this.optService.getProvincias();
        this.isLoading = false;
      } else if (componentInstance instanceof FormAreasDeIncumbenciaComponent) {
        // FORM AREAS DE INCUMBENCIA
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.AREAS_INCUMBENCIA;
        this.editTittle = this.titles.datosAreas;
        this.actualCategory = this.categorys.REQUERIDO;
        componentInstance.areas = await this.areasService.getAreas();
        this.isLoading = false;
      } else if (componentInstance instanceof FormDatosCvNivelEducativoComponent) {
        // FORM NIVEL EDUCATIVO
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.NIVEL_EDUCATIVO;
        this.editTittle = this.titles.datosEducacion;
        this.actualCategory = this.categorys.ADICIONAL;
        componentInstance.nivelesEducativosBack = await this.optService.getNivelEducativo();
        this.isLoading = false;
      } else if (componentInstance instanceof FormExperienciaLaboralComponent) {
        // FORM EXPERIENCIA LABORAL
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.EXPERIENCIA_LABORAL;
        this.editTittle = this.titles.datosExperiencia;
        this.actualCategory = this.categorys.ADICIONAL;
        this.isLoading = false;
      } else if (componentInstance instanceof FormInformaticaComponent) {
        // FORM INFORMATICA
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.INFORMATICA;
        this.editTittle = this.titles.datosInformatica;
        this.actualCategory = this.categorys.ADICIONAL;
        this.isLoading = false;
      } else if (componentInstance instanceof FormDatosCvCursoComponent) {
        // FORM CURSO
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.CURSO;
        this.editTittle = this.titles.datosCurso;
        this.actualCategory = this.categorys.ADICIONAL;
        this.isLoading = false;
      } else if (componentInstance instanceof FormDatosCvIdiomaComponent) {
        // FORM IDIOMA
        this.urlReload = ROUTES.EDITAR + "/" + ROUTES.IDIOMA;
        this.editTittle = this.titles.datosIdioma;
        this.actualCategory = this.categorys.ADICIONAL;
        componentInstance.idiomasBack = await this.optService.getIdioma();
        this.isLoading = false;
      }
    } catch (error) {
      this.isLoading = false;
      this.httpError = true;
    }
    this.actualComponentInstance = componentInstance;
  }

  /**
   * Si los datos del formulario estan validados, intenta 
   */
  async actualizarInfo() {
    // Le solicita al componente instanciado (formulario) que actualice la información, devuelve un boolean.
    if (this.actualComponentInstance.actualizarInfo()) {
      try {
        //Se actualizan los datos del postulante y se muestra la pantalla de resumen de datos
        if (this.actualCategory === this.categorys.REQUERIDO) {
          this.perfilPostulanteService.actualizarPerfilPostulante();
        } else if (this.actualCategory === this.categorys.ADICIONAL) {
          switch (this.editTittle) {
            case this.titles.datosEducacion:
              await this.datosCVService.actualizarEducacionPostulante();
              break;
            case this.titles.datosCurso:
              await this.datosCVService.actualizarCursoPostulante();
              break;
            case this.titles.datosExperiencia:
              await this.datosCVService.actualizarExpLaboralPostulante();
              break;
            case this.titles.datosIdioma:
              await this.datosCVService.actualizarIdiomaPostulante();
              break;
            case this.titles.datosInformatica:
              await this.datosCVService.actualizarInformaticaPostulante();
              break;
            default:
              console.log("Entro default")
              break;
          }
        }
        //Redirecciona utilizando el router hacia el resumen 
        this.router.navigate([ROUTES.RESUMEN]);
      } catch (error) {
        this.httpError = true;
      }
    }
  }

  /**
   * Redirecciona utilizando el router hacia el resumen 
   */
  regresar() {
    this.router.navigate([this.routes.RESUMEN]);
  }
}
