import { Component, OnInit } from '@angular/core';
import { Observable } from 'rxjs';
import { PostulanteService } from 'src/app/servicios/postulante.service';

@Component({
  selector: 'app-footer',
  templateUrl: './footer.component.html',
  styleUrls: ['./footer.component.css']
})
export class FooterComponent implements OnInit {
  idUsuario: number = 6;
  message: string = '';
  userExist: boolean = false;

  tienePerfil$: Observable<boolean>;

  constructor(private postulanteService: PostulanteService) {
    this.tienePerfil$ = postulanteService.tienePerfil.asObservable()
    if(localStorage.getItem('userIdStorate'))
      this.idUsuario = parseInt(localStorage.getItem('userIdStorate'))
  }

  ngOnInit(): void {
    this.postulanteService.obtenerPostulanteByIdUsuario(this.idUsuario)
  }

  verifyUser() {
    localStorage.setItem('userIdStorate',JSON.stringify(this.idUsuario))
    this.postulanteService.obtenerPostulanteByIdUsuario(this.idUsuario)
  }

}
