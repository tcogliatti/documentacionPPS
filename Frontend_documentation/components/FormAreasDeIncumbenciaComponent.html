import { Component, Input, OnInit, ViewChild } from '@angular/core';
import { Area } from 'src/app/interfaces/Area';
import { SelectAutocompleteComponent } from '../select-autocomplete/select-autocomplete.component';
import { Observable, Subscription } from 'rxjs';
import { PostulantePerfil } from 'src/app/interfaces/PostulantePerfil';
import { PostulanteService } from 'src/app/servicios/postulante.service';

@Component({
  selector: 'app-form-areas-de-incumbencia',
  templateUrl: './form-areas-de-incumbencia.component.html',
  styleUrls: ['./form-areas-de-incumbencia.component.css']
})

/**
 * Componente para la selección y carga de las areas de incumbencia del postulante.
 * Recibe del comopnente padre un array con las areas de incumbencia
 */
export class FormAreasDeIncumbenciaComponent implements OnInit {
  /**
 * Instacia del SelectAtucompleteComponent
 */
  @ViewChild(SelectAutocompleteComponent) childComponent!: SelectAutocompleteComponent;


  /**
 * Arreglo de items del tipo Area seleccionadas por el usuario
 */
  areasElegidas: Area[] = [];

  /**
   * Areas que se obtendran del backend para que el usuario las seleccione
   */
  @Input()
  areas: Area[] = [];

  /**
   * Area que seleccionara el usuario en el selectAtucomplete
   */
  areaActual!: Area | null;
  /**
   * Estado local del perfil del postulante 
   */
  postulanteForm!: PostulantePerfil;
  /**
 * Declaración del observable y la suscripción para conectar con postulante service
 */
  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;


  constructor(private postulanteService: PostulanteService) {
    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable()
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(postulanteFromService => {
      this.postulanteForm = { ...postulanteFromService };
      this.areasElegidas = [...this.postulanteForm.areasIncumbencia];
    })
  }

  ngOnInit(): void {
  }

  /**
 * Limpia el componente hijo SelectAutocomplete
 */
  clearChild(): void {
    this.childComponent.clear();
  }

  /**
   * Agrega localmente las areas de incumbencia que seleccione el usuario
   */
  agregar($event: Area) {
    if (this.areasElegidas.length > 0) {
      let select = document.querySelector("#areas-de-incumbencia");
      if (select) {
        let inputGenero: HTMLInputElement | null = select?.querySelector('input');
        if (inputGenero) {
          inputGenero.classList.remove("is-invalid")
        }
      }
      select?.classList.remove("is-invalid")
    }

    /**
     * Cuando se emite el evento datoChange puede ser porque el usuario selecciono una tarea o
     * cuando limpiamos el componente selectAutocomplete, asi que debemos chequear si el parametro $event
     * es null cortar la ejecucion de la funcion
    */
    if ($event === null) {
      console.log("Selecciono nada.", $event);
      return
    }

    const select: Area = { ...$event };

    let areaObj: Area | undefined = this.areas.find((item: Area) => item.idarea === select.idarea);

    //Checkea que lo seleccionado por el usuario sea valido
    if (areaObj === undefined) {
      return
    }

    //Checkea si el area ya estaba seleccionada
    if (this.areasElegidas.find((item) => areaObj?.descripcion == item.descripcion)) {
      this.clearChild();
      return
    }

    //Si todo salio bien agregamos el Area a nuestras areasElegidas y limpiamos el componente
    this.areasElegidas.push(areaObj);
    this.clearChild();
  }

  /**
   * Desagrega localmente el area de incumbencia seleccionada
   */
  desagregar(area: Area) {
    this.areasElegidas = this.areasElegidas.filter((item) => item.idarea != area.idarea);
  }

  /**
 * Esta funcion se encargara de guardar (haciendo uso del service del postulante),
 * el atributo "this.areasElegidas".
 */
  actualizarInfo(): boolean {
    //Verificamos que areaselegidas no sea vacio
    if (this.areasElegidas.length === 0) {
      let select = document.querySelector("#areas-de-incumbencia");
      if (select) {
        let inputGenero: HTMLInputElement | null = select?.querySelector('input');
        if (inputGenero) {
          inputGenero.classList.add("is-invalid")
        }
      }
      select?.classList.add("is-invalid")
      return false;
    }

    this.updatePostualanteLocal();

    return true;

  }
  /**
 * Actualiza el observable y avisa al postulanteService del cambio
 */
  updatePostualanteLocal() {
    this.postulanteForm.areasIncumbencia = this.areasElegidas;
    this.postulanteService.updateEstadoPostulanteLocal(this.postulanteForm);
  }
}

<div class="container d-flex flex-column justify-content-center my-5">
    <div class="mb-2 mx-auto d-flex flex-column input-areas-container">
        <label for="areas-de-incumbencia" class="form-label m-0">Áreas de incumbencia <span
                class="obligatorio">*</span></label>
        <app-select-autcomplete class="input-areas select-requerido" [(dato)]="areaActual"
            (datoChange)="agregar($event)" [collection]=areas id="areas-de-incumbencia"></app-select-autcomplete>

        <div class="invalid-feedback m-0">
            Por favor seleccione al menos un área de interés
        </div>
    </div>
    <ul class="my-3 mx-auto lista-agregadas list-group list-group-flush">
        <div *ngIf="areasElegidas.length > 0">
            <p class="mb-1">Áreas de incumbencia seleccionadas</p>
            <div *ngFor="let area of areasElegidas">
                <li class=" my-2 list-group-item list-group-item-action d-flex justify-content-between align-items-center gap-2"
                    value="area.idarea">
                    {{area.descripcion}} <button (click)="desagregar(area)"
                        class="btn btn-desagregar btn-danger rounded-circle"><i
                            class="bi bi-sm bi-trash-fill"></i></button>
                </li>
            </div>
        </div>
    </ul>

</div>
