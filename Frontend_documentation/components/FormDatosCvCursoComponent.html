import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
import { Curso } from 'src/app/interfaces/Curso';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';

@Component({
  selector: 'app-form-datos-cv-curso',
  templateUrl: './form-datos-cv-curso.component.html',
  styleUrls: ['./form-datos-cv-curso.component.css'],
})

/**
 * Formulario para carga de cursos del postulante
 */
export class FormDatosCvCursoComponent implements OnInit {
  // id inicial del curso actual 
  id: number = 0;
  // variable para guardar el id del ultimo curso agregado 
  ultimoID: number = 0;
  // arreglo de anios para mostrar en el select 
  anios: number[] = [];

  // flag que cambia el titulo entre "agregar curso" y "editar curso"
  editandoFlag: boolean = false;

  cursos: Curso[] = [];

  /**
   * este es el objeto del curso actual, el cual se actualiza por medio de :
   *  two way data binding al actualizar los valores de los inputs
   */
  cursoActual: Curso = {
    idcurso: NaN,
    idpostulante: 0,
    descripcion: '',
    establecimiento: '',
    duracion: '',
    anio: 0,
    estado: '',
  };
  
  datosCVObservable$!: Observable<DatosAdicionales>;
  datosCVSuscription!: Subscription;
  datosCVForm!: DatosAdicionales;

  constructor(private router: Router,private datosCVService: DatosAdicionalesService) {     /* DATOS CV SERVICE */
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = JSON.parse(JSON.stringify(datosCVFormService));
      this.cursos = JSON.parse(JSON.stringify(this.datosCVForm.curso));
    });
  }

  
  ngOnInit(): void {
    /* se llena el arreglo de completar anios */
    this.anios = this.completarArregloAnios();
    
  }

 /**
  * Actualiza el observable y avisa al postulanteService del cambio
  */
 updateDatosCVLocal() {
  this.datosCVForm.curso = this.cursos;
  this.datosCVService.updateEstadoDatosCVLocal(this.datosCVForm);
}
  
  actualizarInfo(): boolean {
    // Se actualizan los datos del cv del postulante
    this.updateDatosCVLocal();
    return true; 
  }

  // eliminar objeto curso de la lista
  eliminar() {
    this.cursos = this.cursos.filter(
      (curso) => curso.idcurso != this.cursoActual.idcurso
    );
  }

  /**
   * realiza las validaciones de boostrap, si los inputs estan vacios siendo requeridos
   * no permite avanzar con el envio del form 
   */
  validar(): any {
    const form: HTMLFormElement | null =
      document.querySelector('#formDatosCurso');
    let esValido = true;

    if (form == null) {
      return false;
    }

    let estado = this.cursoActual.estado.toLowerCase();
    // Validamos que no seleccione otra cosa que no sea las que deseamos
    if (
      estado != 'incompleto' &&
      estado != 'en curso' &&
      estado != 'completo'
    ) {
      esValido = false;
      this.cursoActual.estado = '';
    }

    // verificamos si ingreso otra cosa que no sea un año
    let anio: any = this.cursoActual.anio;
    if (anio == 'Seleccionar un año') {
      esValido = false;
      this.cursoActual.anio = NaN;
    }

    /* si todos los inputs requeridos estan completos se agrega el curso y se cierra el modal */
    if (form.checkValidity() && esValido) {
      if (form.classList.contains('was-validated')) {
        //Normalizamos el formulario
        form.classList.remove('was-validated');
      }
      //forzamos el cierre del form si todo salio bien
      this.cerrarModal();
      return true;
    } else {
      form.classList.add('was-validated');
      return false;
    }
  }

  /**
   * crea un nuevo curso con los datos del formulario
    luego verifica que el valor del select sea valido 
    y luego pushea el nuevo curso al arreglo de cursos
    finalmente actualiza el id del ultimo curso agregado para guardar su valor
   */
  agregarCurso() {
    if (!this.validar()) {
      return;
    }
    const nuevoCurso: Curso = {
      idcurso: this.ultimoID + 1,
      idpostulante: 0,
      descripcion: this.cursoActual.descripcion.toLocaleLowerCase(),
      establecimiento: this.cursoActual.establecimiento.toLocaleLowerCase(),
      duracion: this.cursoActual.duracion.toLocaleLowerCase(),
      anio: this.cursoActual.anio,
      estado: this.cursoActual.estado.toLocaleLowerCase(),
    };
    if (nuevoCurso.idcurso)
      this.ultimoID = nuevoCurso.idcurso;

    this.cursos.push(nuevoCurso);
  }

 /**
  * actualiza el curso actual a vacio
  */
  vaciarForm() {
    this.cursoActual = {
      idcurso: NaN,
      idpostulante: 0,
      descripcion: '',
      establecimiento: '',
      duracion: '',
      anio: 0,
      estado: '',
    };
  }

  /* primero cambia a true el flag que muestra el titulo de "edicion de curso" 
     luego actualiza el curso actual al que llega por parametro */
  abrirEditar(curso: Curso) {
    this.editandoFlag = true;
    this.actualizarCursoLocal(curso);
  }

  /*  primero busca en el arreglo de curso si existe algun curso con el id del curso actual y lo guarda
      luego pisa ese curso guardado con la informacion del curso actual (los datos del form de edicion)
  */
  actualizarCurso() {
    if (!this.validar()) {
      return;
    }

    let cursoAactualizar: Curso | undefined = this.cursos.find(
      (item) => item.idcurso === this.cursoActual.idcurso
    );
    if (cursoAactualizar == undefined) {
      return;
    }
    (cursoAactualizar.idcurso = this.cursoActual.idcurso),
      (cursoAactualizar.idpostulante = this.cursoActual.idpostulante),
      (cursoAactualizar.descripcion =
        this.cursoActual.descripcion.toLocaleLowerCase()),
      (cursoAactualizar.establecimiento =
        this.cursoActual.establecimiento.toLocaleLowerCase()),
      (cursoAactualizar.duracion =
        this.cursoActual.duracion.toLocaleLowerCase()),
      (cursoAactualizar.anio = this.cursoActual.anio),
      (cursoAactualizar.estado = this.cursoActual.estado.toLocaleLowerCase()),
      this.cerrarModal();
  }

  /**
    Actualiza el objeto enlazado al formulario con la informacion que venga por parametros
    @param curso CursoPostulante
  */
  actualizarCursoLocal(curso: Curso) {
    // Clonar el objeto educacionPostulanteActual y asignar correctamente el objeto niveleducativo
    this.cursoActual = {
      idcurso: curso.idcurso,
      idpostulante: curso.idpostulante,
      descripcion: curso.descripcion.toLocaleLowerCase(),
      establecimiento: curso.establecimiento.toLocaleLowerCase(),
      duracion: curso.duracion.toLocaleLowerCase(),
      anio: curso.anio,
      estado: curso.estado.toLocaleLowerCase(),
    };
  }

  /* primero vacia el form de carga para que los inputs esten vacios
    y luego cambia a false el flag que muestra el titulo de "edicion de curso" */
  abrirFormCarga() {
    this.vaciarForm();
    this.editandoFlag = false;
    this.normalizarForm();
  }

  /* metodo que retorna en año actual */
  getAnioActual(): number {
    let fecha = new Date();
    let año = fecha.getFullYear();
    return año;
  }

  /**
   * metodo encargado de completar el arreglo de años que se utiliza en el select de año
   *
   */
  completarArregloAnios(): number[] {
    let anioActual = this.getAnioActual();
    let respuesta: number[] = [];
    /* se pushea al arreglo de anios la cantidad de años desde el actual hasta 50 años para atras */
    for (let i = anioActual - 50; i <= anioActual; i++) {
      respuesta.push(i);
    }
    respuesta.reverse()
    return respuesta;
  }

  // cerrar modal una vez que todos los inputs fueron validados
  cerrarModal() {
    const btnClose = document.getElementById('btn-close');
    if (btnClose) {
      btnClose.click();
    }
  }

  private normalizarForm() {
    const form: HTMLFormElement | null =
      document.querySelector('#formDatosCurso');

    if (form != null) {
      if (form.classList.contains('was-validated')) {
        form.classList.remove('was-validated');
      }
    }
  }
}
