import { Component, Input, OnInit } from '@angular/core';
import { Observable, Subscription } from 'rxjs';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { Idioma } from 'src/app/interfaces/Idioma';
import { IdiomaPostulante } from 'src/app/interfaces/IdiomaPostulante';
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';

@Component({
  selector: 'app-form-datos-cv-idioma',
  templateUrl: './form-datos-cv-idioma.component.html',
  styleUrls: ['./form-datos-cv-idioma.component.css']
})

/**
 * Componente que contiene el formualrio de carga de idiomas y el registro de los idiomas cargados
 * Recibe del componente padre un array con los idiomas
 */
export class FormDatosCvIdiomaComponent implements OnInit {
  /**
   * Guarda una lista de objetos tipo IdiomaPostulante:
   * 
   * Opcion A: Es la primera ves que se carga la info y estara vacio = []
   * 
   * Opcion B: Ya posee informacion guardada y se obtendra del servicio CvPostulanteServi
   */
  idiomaPostulante: IdiomaPostulante[] = [];

  /**
   * Arreglo de objetos del tipo : Idioma que servira para manejar las opciones que pueda seleccionar el usuario
   * el cual se instaciara a partir del arreglo de objetos "idiomasBack [] : Idioma"   
   */
  // idiomas: Idioma[] = [];
  anios: number[] = [];


  /**
   * Arreglo de objetos del tipo : Idioma que provendra desde de la BBDD
   */
  @Input()
  idiomasBack: Idioma[] = [];

  /**
   * Objeto bindiado con el formulario, el cual servira para manejar las distintas acciones del usuario
   */
  idiomaPostulanteActual: IdiomaPostulante = {
    ididioma: { descripcion: "", ididioma: NaN },
    idpostulante: NaN,
    nivel: "",
    estado: "",
    anio: ""
  }

  /**
   * Flag usada para renderizar condicionalmente los detalles del formulario entre "Editar" y "Agregar"
   */
  editandoFlag: boolean = false;

  /**
   * Datos CV Servicio
   */
  datosCVObservable$!: Observable<DatosAdicionales>;
  datosCVSuscription!: Subscription;
  datosCVForm!: DatosAdicionales;

  constructor(private datosCVService: DatosAdicionalesService) {     /* DATOS CV SERVICE */
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = JSON.parse(JSON.stringify(datosCVFormService));
      this.idiomaPostulante = JSON.parse(JSON.stringify(this.datosCVForm.idioma));
    });
  }

  ngOnInit(): void {
    /* se llena el arreglo de completar anios */
    this.anios = this.completarArregloAnios();

  }

  /**
  * Actualiza el observable y avisa al postulanteService del cambio
  */
  updateDatosCVLocal() {
    this.datosCVForm.idioma = this.idiomaPostulante;
    this.datosCVService.updateEstadoDatosCVLocal(this.datosCVForm);
  }

  /**
   * TODO: Cuando el componente se separe, esta debera actualizar la info de su hijo(FormNivelEducativo). Y si todo
   * sale bien reien ahi navegar hacia cursos
   */
  actualizarInfo(): boolean {
    // Se actualizan los datos del cv del postulante
    this.updateDatosCVLocal();
    return true;
  }

  /**
   * Elimina el idioma de la lista
   */
  eliminar() {
    this.idiomaPostulante = this.idiomaPostulante.filter(item => item.ididioma.ididioma !== this.idiomaPostulanteActual.ididioma.ididioma);
  }

  /**
   * Blanquea el formulario enlazado al idiomaPostulanteActual y setea el editar como falso
   */
  abrirAgregar(): void {
    this.editandoFlag = false;
    this.normalizarForm();

    this.blanquearForm();
  }

  private normalizarForm() {
    const form: HTMLFormElement | null = document.querySelector("#formIdioma");

    if (form != null) {
      if (form.classList.contains("was-validated")) {
        form.classList.remove("was-validated");
      }
    }
  }

  /**
   * Si pasan las validaciones, clonaremos la informacion cargada del usuario y la agregamos al arreglo local
   * "idiomas", luego filtraremos el idioma seleccionado por el usuario para que no lo elija 
   * dos veces. Por ultimo se blanquea el formulario.
   */
  agregarIdioma() {

    if (!this.validar()) {
      return
    }

    // Clonar el objeto idiomaPostulanteActual y asignar correctamente el objeto idioma
    const nuevoIdiomaPostulante: IdiomaPostulante = {
      ididioma: {
        descripcion: this.idiomaPostulanteActual.ididioma.descripcion,
        ididioma: this.idiomaPostulanteActual.ididioma.ididioma
      },
      nivel: this.idiomaPostulanteActual.nivel.toLowerCase(),
      anio: this.idiomaPostulanteActual.anio,
      estado: this.idiomaPostulanteActual.estado.toLowerCase(),
      idpostulante: this.idiomaPostulanteActual.idpostulante,
    };

    // Encontrar la posición donde debe ser insertado el nuevo elemento
    let posicion: number = this.idiomaPostulante.findIndex(elemento => elemento.ididioma.ididioma > nuevoIdiomaPostulante.ididioma.ididioma);
    // Si no se encuentra una posición mayor, significa que el nuevo elemento es el más grande
    if (posicion === -1) {
      posicion = this.idiomaPostulante.length;
    }

    // Agregamos el nuevo objeto al arreglo local
    this.idiomaPostulante.splice(posicion, 0, nuevoIdiomaPostulante);

    // Blanqueamos el formulario
    this.blanquearForm();
  }

  /**
   * Activa la bandera de edicion y actualiza el formulario con la info de la card (param item)
   * @param item IdiomaPostulante
   */
  abrirEditar(item: IdiomaPostulante): void {
    // Seteamos el form en modo editar
    this.editandoFlag = true;
    console.log(item);

    // Actualizamos los input del form con la info de la card
    this.actualizarIdioma(item);
    // Normalizamos el form
    this.normalizarForm();
  }

  /**
   * Si pasan las validaciones, buscamos el objeto enlazado a la card en el arreglo "idioma"
   * Si lo encontramos actualizamos la info de la card con la nueva informacion del formulario. Por ultimo
   * cerramos la bandera de edicion y blanqueamos el formulario.
   */
  editarIdioma(): void {

    if (!this.validar()) {
      return
    }

    // Buscamos el objeto enlazado a la card segun el ididioma
    let item: IdiomaPostulante | undefined = this.idiomaPostulante.find(item => item.ididioma.ididioma === this.idiomaPostulanteActual.ididioma.ididioma)

    if (item === undefined) {
      console.log("Ocurrio un error al editar.")
      return
    }

    // Clonar el objeto idiomaPostulanteActual y asignar correctamente el objeto idioma
    item.ididioma.descripcion = this.idiomaPostulanteActual.ididioma.descripcion;
    item.ididioma.ididioma = this.idiomaPostulanteActual.ididioma.ididioma;
    item.nivel = this.idiomaPostulanteActual.nivel.toLowerCase();
    item.anio = this.idiomaPostulanteActual.anio;
    item.estado = this.idiomaPostulanteActual.estado.toLowerCase();
    item.idpostulante = this.idiomaPostulanteActual.idpostulante;

    // Cerramos la bandera de edicion
    this.editandoFlag = false;
    // Blanqueamos el formulario
    this.blanquearForm();
  }

  /**
   * Valida el formulario de idioma tanto por bootstrap como validaciones manuales
   * @returns boolean
   */
  validar(): boolean {
    const form: HTMLFormElement | null = document.querySelector("#formIdioma");
    let esValido = true;

    if (form == null) {
      return false
    }

    //Validamos que el idioma sea correcto
    if (!this.idiomasBack.find(item => item.descripcion === this.idiomaPostulanteActual.ididioma.descripcion)) {
      esValido = false;
      this.idiomaPostulanteActual.ididioma = { descripcion: "", ididioma: NaN }
    }

    //Validamos que el estado sea correcto
    let estado = this.idiomaPostulanteActual.estado.toLowerCase();
    if (estado != "incompleto" && estado != "en curso" && estado != "completo") {
      esValido = false;
      this.idiomaPostulanteActual.estado = ""
    }
    //Validamos por Bootstrap y checkeamos las validaciones anteriores
    if (form.checkValidity() && esValido) {
      //Si paso y antes se habia validado
      if (form.classList.contains("was-validated")) {
        //Normalizamos el formulario
        form.classList.remove("was-validated")
      }
      //forzamos el cierre del form si todo salio bien
      this.cerrarForm();
      return true;
    }

    //Si no paso validamos la interface
    form.classList.add("was-validated");
    return false;
  }

  /**
   * Actualiza el objeto enlazado al formulario con informacion vacia
   */
  blanquearForm() {
    this.idiomaPostulanteActual = {
      ididioma: { descripcion: "", ididioma: NaN },
      nivel: "",
      anio: "",
      estado: "",
      idpostulante: NaN,
    };
  }

    /**
    * Actualiza el objeto enlazado al formulario con la informacion que venga por parametros
    * @param item del tipo IdiomaPostulante
    */
  actualizarIdioma(item: IdiomaPostulante) {
    console.log(item);
    // Clonar el objeto idiomaPostulanteActual y asignar correctamente el objeto idioma
    this.idiomaPostulanteActual = {
      ididioma: {
        descripcion: item.ididioma.descripcion,
        ididioma: item.ididioma.ididioma
      },
      nivel: item.nivel.toLowerCase(),
      anio: item.anio,
      estado: item.estado.toLowerCase(),
      idpostulante: item.idpostulante,
    };
    console.log(this.idiomaPostulanteActual);

  }



  /**
  * Función de comparación por id dado dos idiomas
  */
  private compararPorId(a: Idioma, b: Idioma) {
    return a.ididioma - b.ididioma;
  }

  /**
   *  Cierre forzado del formulario modal
   */
  cerrarForm(): void {
    const buttonCerrar: HTMLButtonElement | null = document.querySelector("#btn-cerrar-form");
    if (buttonCerrar == null) {
      return
    }
    buttonCerrar.click();
  }

  /* metodo encargado de completar el arreglo de años que se utiliza en el select de año */
  completarArregloAnios(): number[] {
    let anioActual = this.getAnioActual();
    let respuesta: number[] = [];
    /* se pushea al arreglo de anios la cantidad de años desde el actual hasta 50 años para atras */
    for (let i = anioActual - 50; i <= anioActual; i++) {
      respuesta.push(i);
    }

    respuesta.reverse()



    return respuesta;
  }

  /* metodo que retorna en año actual */
  getAnioActual(): number {
    let fecha = new Date();
    let año = fecha.getFullYear();
    return año;
  }

}
