import { Component, Input, OnInit } from '@angular/core';
import { Observable, Subscription } from 'rxjs';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { EducacionPostulante } from 'src/app/interfaces/EducacionPostulante';
import { NivelEducativo } from 'src/app/interfaces/NivelEducativo';
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';

@Component({
  selector: 'app-form-datos-cv-nivel-educativo',
  templateUrl: './form-datos-cv-nivel-educativo.component.html',
  styleUrls: ['./form-datos-cv-nivel-educativo.component.css']
})

/**
 * Componente que contiene el formulario de carga de nivel educativo y los nieveles educativos que el postulante ha cargado
 * Recibe del componente padre un array con los niveles educativos
 */
export class FormDatosCvNivelEducativoComponent {
  /**
   * Guarda una lista de objetos tipo EducacionPostulante:
   * 
   * Opcion A: Es la primera ves que se carga la info y estara vacio = []
   * 
   * Opcion B: Ya posee informacion guardada y se obtendra del servicio CvPostulanteServi
   */
  educacionPostulante: EducacionPostulante[] = [];

  /**
   * Arreglo de objetos del tipo : NivelEducativo que servira para manejar las opciones que pueda seleccionar el usuario
   * el cual se instaciara a partir del arreglo de objetos "nivelesEduactivosBack [] : NivelEducativo"   
   */
  nivelesEducativos: NivelEducativo[] = [];

  /**
   * Arreglo de objetos del tipo : NivelEducativo que provendra desde de la BBDD
   */
  @Input()
  nivelesEducativosBack: NivelEducativo[] = [];

  /**
   * Objeto bindiado con el formualariom, el cual servira para manejar las distintas acciones del usuario
   */
  educacionPostulanteActual: EducacionPostulante = {
    idniveleducativo: { descripcion: "", idniveleducativo: NaN },
    especialidad: "",
    establecimiento: "",
    estado: "",
    idpostulante: NaN,
  }

  /**
   * Flag usada para renderizar condicionalmente los detalles del formulario entre "Editar" y "Agregar"
   */
  editandoFlag: boolean = false;

  /**
   * Datos CV Servicio
   */
  datosCVObservable$!: Observable<DatosAdicionales>;
  datosCVSuscription!: Subscription;
  datosCVForm!: DatosAdicionales;

  constructor(private datosCVService: DatosAdicionalesService) {     /* DATOS CV SERVICE */
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = JSON.parse(JSON.stringify(datosCVFormService));
      this.educacionPostulante = JSON.parse(JSON.stringify(this.datosCVForm.educacionPostulante));
    });
  }


  /**
  * Actualiza el observable y avisa al postulanteService del cambio
  */
  updateDatosCVLocal() {
    this.datosCVForm.educacionPostulante = this.educacionPostulante;
    this.datosCVService.updateEstadoDatosCVLocal(this.datosCVForm);
  }

  /**
   * Intenta actualizar la informacion del DatosAdicionalesService y devuelve true.
   */
  actualizarInfo(): boolean {
    // Se actualizan los datos del cv del postulante
    this.updateDatosCVLocal();
    return true;
  }

  /**
   * Filtra el arreglo local de "educacionPostulante", y actualiza el arreglo local de nivelesEducativos.
   * Por ultimo lo ordena segun el id haciendo uso interno de la funcion "compararPorId()"
   */
  eliminar() {
    this.nivelesEducativos.push(JSON.parse(JSON.stringify(this.educacionPostulanteActual.idniveleducativo)));
    this.educacionPostulante = this.educacionPostulante.filter(item => item.idniveleducativo.idniveleducativo !== this.educacionPostulanteActual.idniveleducativo.idniveleducativo)

    // Ordenar el arreglo
    this.nivelesEducativos.sort(this.compararPorId);
  }

  /**
   * Setea el editar como falso y blanquea el formulario enlazado al educacionPostulanteActual
   */
  abrirAgregar(): void {
    this.editandoFlag = false;
    this.normalizarForm();
    // Si es la primera entrada a la app y todavia hay niveles por cargar 
    if (this.nivelesEducativos.length === 0 && this.educacionPostulante.length !== this.nivelesEducativosBack.length) {
      this.nivelesEducativos = JSON.parse(JSON.stringify(this.nivelesEducativosBack));
      // Si existen niveles eduacativos cargados, filtramos las opciones de niveles eduactivos que no se hayan seleccionado
      if (this.educacionPostulante.length != 0) {
        this.educacionPostulante.forEach(eduPostu => {
          this.nivelesEducativos = this.nivelesEducativos.filter(nivel => nivel.idniveleducativo !== eduPostu.idniveleducativo.idniveleducativo)
        })
      }
    }

    this.blanquearForm();
  }

  /**
   * Desactiva la validacion del formulario si es que ya fue validado
   * removiendo las clases de bootstrap para la validacion.
   */
  private normalizarForm() {
    const form: HTMLFormElement | null = document.querySelector("#formNivelEducativo");

    if (form != null) {
      if (form.classList.contains("was-validated")) {
        form.classList.remove("was-validated");
      }
    }
  }

  /**
   * Si pasan las validaciones, clonaremos la informacion cargada del usuario y la agregamos al arreglo local
   * "nivelesEducativos", luego filtraremos el nivel educativo seleccionado por el usuario para que no lo elija 
   * dos veces. Por ultimo se blanquea el formulario.
   */
  agregarNivel() {

    if (!this.validar()) {
      return
    }

    // Clonar el objeto educacionPostulanteActual y asignar correctamente el objeto niveleducativo
    const nuevoEducacionPostulante = {
      idniveleducativo: {
        descripcion: this.educacionPostulanteActual.idniveleducativo.descripcion,
        idniveleducativo: this.educacionPostulanteActual.idniveleducativo.idniveleducativo
      },
      especialidad: this.educacionPostulanteActual.especialidad.toLowerCase(),
      establecimiento: this.educacionPostulanteActual.establecimiento.toLowerCase(),
      estado: this.educacionPostulanteActual.estado.toLowerCase(),
      idpostulante: this.educacionPostulanteActual.idpostulante,
    };

    // Filtramos las opciones del nivel educativo para que no carge 2 veces el mismo 
    this.nivelesEducativos = this.nivelesEducativos.filter(item => item.idniveleducativo !== nuevoEducacionPostulante.idniveleducativo.idniveleducativo);

    // Encontrar la posición donde debe ser insertado el nuevo elemento
    let posicion: number = this.educacionPostulante.findIndex(elemento => elemento.idniveleducativo.idniveleducativo > nuevoEducacionPostulante.idniveleducativo.idniveleducativo);
    // Si no se encuentra una posición mayor, significa que el nuevo elemento es el más grande
    if (posicion === -1) {
      posicion = this.educacionPostulante.length;
    }

    // Agregamos el nuevo objeto al arreglo local
    this.educacionPostulante.splice(posicion, 0, nuevoEducacionPostulante);

    // Blanqueamos el formulario
    this.blanquearForm();
  }

  /**
   * Activa la bandera de edicion y actualiza el formulario con la info de la card (param item)
   * @param item EducacionPostulante
   */
  abrirEditar(item: EducacionPostulante): void {
    // Seteamos el form en modo editar
    this.editandoFlag = true;
    // Actualizamos los input del form con la info de la card
    this.actualizarEducacionPostuLocal(item);
    // Normalizamos el form
    this.normalizarForm();
  }

  /**
   * Si pasan las validaciones, buscamos el objeto enlazado a la card en el arreglo "educacionPostulante"
   * Si lo encontramos actualizamos la info de la card con la nueva informacion del formulario. Por ultimo
   * cerramos la bandera de edicion y blanqueamos el formulario.
   */
  editarNivel(): void {

    if (!this.validar()) {
      return
    }

    // Buscamos el objeto enlazado a la card segun el idniveleducativo
    let item: EducacionPostulante | undefined = this.educacionPostulante.find(item => item.idniveleducativo.idniveleducativo === this.educacionPostulanteActual.idniveleducativo.idniveleducativo)

    if (item === undefined) {
      console.log("Ocurrio un error al editar.")
      return
    }

    // Clonar el objeto educacionPostulanteActual y asignar correctamente el objeto niveleducativo
    item.idniveleducativo.descripcion = this.educacionPostulanteActual.idniveleducativo.descripcion;
    item.idniveleducativo.idniveleducativo = this.educacionPostulanteActual.idniveleducativo.idniveleducativo;
    item.especialidad = this.educacionPostulanteActual.especialidad.toLowerCase();
    item.establecimiento = this.educacionPostulanteActual.establecimiento.toLowerCase();
    item.estado = this.educacionPostulanteActual.estado.toLowerCase();
    item.idpostulante = this.educacionPostulanteActual.idpostulante;

    // Cerramos la bandera de edicion
    this.editandoFlag = false;
    // Blanqueamos el formulario
    this.blanquearForm();
  }

  /**
   * Valida el formulario de nivel educativo tanto por bootstrap como validaciones manuales
   * @returns boolean
   */
  validar(): boolean {
    const form: HTMLFormElement | null = document.querySelector("#formNivelEducativo");
    let esValido = true;

    if (form == null) {
      return false
    }

    //Validamos que el nivel educativo sea correcto
    if (!this.nivelesEducativosBack.find(item => item.descripcion === this.educacionPostulanteActual.idniveleducativo.descripcion)) {
      esValido = false;
      this.educacionPostulanteActual.idniveleducativo = { descripcion: "", idniveleducativo: NaN }
    }

    //Validamos que el estado sea correcto
    let estado = this.educacionPostulanteActual.estado.toLowerCase();
    if (estado != "incompleto" && estado != "en curso" && estado != "completo") {
      esValido = false;
      this.educacionPostulanteActual.estado = ""
    }
    //Validamos por Bootstrap y checkeamos las validaciones anteriores
    if (form.checkValidity() && esValido) {
      //Si paso y antes se habia validado
      if (form.classList.contains("was-validated")) {
        //Normalizamos el formulario
        form.classList.remove("was-validated")
      }
      //forzamos el cierre del form si todo salio bien
      this.cerrarForm();
      return true;
    }

    //Si no paso validamos la interfas
    form.classList.add("was-validated");
    return false;
  }

  /**
   * Actualiza el objeto enlazado al formulario con informacion vacia
   */
  blanquearForm() {
    this.educacionPostulanteActual = {
      idniveleducativo: { descripcion: "", idniveleducativo: NaN },
      especialidad: "",
      establecimiento: "",
      estado: "",
      idpostulante: NaN,
    };
  }

  /**
   * Actualiza el objeto enlazado al formulario con la informacion que venga por parametros
   * @param item EducacionPostulante
   */
  actualizarEducacionPostuLocal(item: EducacionPostulante) {
    // Clonar el objeto educacionPostulanteActual y asignar correctamente el objeto niveleducativo
    this.educacionPostulanteActual = {
      idniveleducativo: {
        descripcion: item.idniveleducativo.descripcion,
        idniveleducativo: item.idniveleducativo.idniveleducativo
      },
      especialidad: item.especialidad.toLowerCase(),
      establecimiento: item.establecimiento.toLowerCase(),
      estado: item.estado.toLowerCase(),
      idpostulante: item.idpostulante,
    };
  }

  /**
  * Función de comparación por id dado dos niveleseducativos
  */
  private compararPorId(a: NivelEducativo, b: NivelEducativo) {
    return a.idniveleducativo - b.idniveleducativo;
  }

  /**
   *  Cierre forzado del formulario modal
   */
  cerrarForm(): void {
    const buttonCerrar: HTMLButtonElement | null = document.querySelector("#btn-cerrar-form");
    if (buttonCerrar == null) {
      return
    }
    buttonCerrar.click();
  }

}
