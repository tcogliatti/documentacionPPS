import { Component, EventEmitter, Input, OnInit, Output, QueryList, ViewChildren } from '@angular/core';
import { MINIMOS, ROUTES } from "../../../constants";
import { Localidad } from "../../interfaces/Localidad";
import { Provincia } from "../../interfaces/Provincia";
import { Calle } from "../../interfaces/Calle";
import { Partido } from "../../interfaces/Partido";
import { Observable, Subscription } from "rxjs";
import { PostulantePerfil } from "../../interfaces/PostulantePerfil";
import { Router } from "@angular/router";
import { PostulanteService } from "../../servicios/postulante.service";
import { ValidarFormHelperService } from "../../servicios/validar-form-helper.service";
import { SelectAutocompleteComponent } from "../select-autocomplete/select-autocomplete.component";

@Component({
  selector: 'app-form-datos-de-contacto',
  templateUrl: './form-datos-de-contacto.component.html',
  styleUrls: ['./form-datos-de-contacto.component.css']
})

/**
 * Contiene el formulario para cargar los datos de contacto
 * Recibe una serie de arrays del componente padre:
 *  1. Provincias
 *  2. Localidades
 *  3. Partido
 *  4. Calle
 */
export class FormDatosDeContactoComponent implements OnInit {
  urlReload: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.DATOS_CONTACTO}`;
  @Input()
  provincias: Provincia[] = [];
  @Input()
  localidades: Localidad[] = [];
  @Input()
  calles: Calle[] = [];
  @Input()
  partidos: Partido[] = [];

  enviado = false; //lo necesita el required del componente ng-autocomplete

  minTel: number = MINIMOS.TELEFONO;
  minCel: number = MINIMOS.CELULAR;
  selectAutoErrorFlag: boolean = false;

  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteForm!: PostulantePerfil;

  @ViewChildren(SelectAutocompleteComponent) childComponents!: QueryList<SelectAutocompleteComponent>;
  constructor(private router: Router, private postulanteService: PostulanteService, private helperValidar: ValidarFormHelperService) {

    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable()
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(postulanteFromService => {
      this.postulanteForm = { ...postulanteFromService }
    })
  }

  async ngOnInit(): Promise<void> {
  }
  /**
   * Método llamado al presionar en el botón de siguiente
   */
  actualizarInfo(): boolean {
    this.enviado = true;

    const form: HTMLFormElement | null = document.querySelector('#formContactData');

    if (form == null) {
      return false;
    }

    if (this.validarForm(form)) {
      // Actualiza el estado del servicio con los nuevos datos
      this.postulanteService.updateEstadoPostulanteLocal(this.postulanteForm);
      return true;

      // Redidige a formulario de datos de contacto
      //this.router.navigate([this.urlFormAreas]);
    } else {
      form.classList.add('was-validated');
      return false;
    }
  }
  private validarForm(form: HTMLFormElement) {
    return this.validar() && form.checkValidity();
  }

  validar(): boolean {
    // Valida que los valores de los selects autocompletar matchen con algun item de su collecion
    this.selectAutoErrorFlag = this.helperValidar.validateAutocompletes(this.childComponents);

    if (this.selectAutoErrorFlag) {
      return false;
    }
    return true;
  }

}
