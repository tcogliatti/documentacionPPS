import { Component, EventEmitter, Input, OnInit, Output, QueryList, ViewChildren } from '@angular/core';
import { Pais } from 'src/app/interfaces/Pais';
import { EstadoCivil } from 'src/app/interfaces/EstadoCivil';
import { TipoDocumento } from 'src/app/interfaces/TipoDocumento';
import { Genero } from 'src/app/interfaces/Genero';
import { API_BASE, MAXIMOS, MINIMOS, ROUTES } from 'src/constants';
import { Router } from '@angular/router';
import { OptionFormService } from 'src/app/servicios/option-form.service';
import { PostulanteService } from 'src/app/servicios/postulante.service';
import { Observable, Subscription } from 'rxjs';
import { PostulantePerfil } from 'src/app/interfaces/PostulantePerfil';
import { SelectAutocompleteComponent } from '../select-autocomplete/select-autocomplete.component';
import { ValidarFormHelperService } from '../../servicios/validar-form-helper.service';

@Component({
  selector: 'app-form-datos-personales',
  templateUrl: './form-datos-personales.component.html',
  styleUrls: ['./form-datos-personales.component.css']
})

/**
 * Componente que contiene el formulario para la carga de los datos personales del postulante
 * Recibe del componente padre:
 *  1. País
 *  2. Tipo de document
 *  3. Estado civil
 *  4. Genero
 */
export class FormDatosPersonalesComponent implements OnInit {

  previewImage!: string | ArrayBuffer | null;

  // url para dirigirse al formulario de contacto
  urlFormContact: string = "/" + ROUTES.DATOS_REQUERIDOS + "/" + ROUTES.DATOS_CONTACTO;

  file: any;

  cuilErrorFlag: boolean = false;
  dniErrorFlag: boolean = false;
  selectAutoErrorFlag: boolean = false;

  @Input()
  paises: Pais[] = [];
  @Input()
  tiposDoc: TipoDocumento[] = [];
  @Input()
  estadosCiviles: EstadoCivil[] = [];
  @Input()
  generos: Genero[] = [];

  isLoading: boolean = true;
  httpError: boolean = false;
  codeandoToBase64: boolean = false;

  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteForm!: PostulantePerfil;

  enviado = false; //lo necesita el required del componente ng-autocomplete

  urlReload: string = `/${ROUTES.DATOS_REQUERIDOS}/${ROUTES.DATOS_PERSONALES}`
  @Output()
  emitUrlReload: EventEmitter<string> = new EventEmitter<string>();

  /**
   * minDate: llama a la funcion calculateDate(con edad minima "18") y retorna la fecha minima a
   * partir de la actualidad para cumplir con la edad minima para poder postularse
   *
   * maxDate: realiza lo mismo pero se le pasa la edad "75" y obtiene la fecha maxima permitida
   *
   * minDate y maxDate son utilizados en el input date "fecha de nacimiento" en los atributos min y max
   */
  fechaNacimiento: any = {
    minDate: this.calculateDate(MINIMOS.EDAD_MINIMA),
    maxDate: this.calculateDate(MAXIMOS.EDAD_MAXIMA),
  }

  @ViewChildren(SelectAutocompleteComponent) childComponents!: QueryList<SelectAutocompleteComponent>;

  constructor(private router: Router, private optFormServ: OptionFormService, private postulanteService: PostulanteService, private helperValidar: ValidarFormHelperService) {

    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable()
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(postulanteFromService => {
    this.postulanteForm = JSON.parse(JSON.stringify(postulanteFromService));
    this.updatePreviewPicture();
    })
  }

  async ngOnInit(): Promise<void> {
  }

  /* ---------------------------------------------- ENVIO Y VALIDACION ----------------------------------------------------  */

  /**
   * Metodo llamado al presionar en el boton de siguiente
  */
  actualizarInfo(): boolean {
    this.enviado = true;

    const form: HTMLFormElement | null = document.querySelector('#formPersonalData');

    if (form == null) {
      return false;
    }

    if (this.validar() && form.checkValidity()) {
      // Actualiza el estado del servicio con los nuevos datos
      this.updatePreviewPicture();
      this.postulanteService.updateEstadoPostulanteLocal(this.postulanteForm);
      return true;
    } else {
      form.classList.add('was-validated');
      return false;
    }
  }

  /**
   * Validar que el usuario no ingrese datos erroneos que bootstrap no puede capturar
   */
  validar(): boolean {
    // Se establecen banderas de control en falso
    this.dniErrorFlag = false;
    this.cuilErrorFlag = false;

    // Valida que el numero de documento contenta como minimo 8 cifras
    if (this.postulanteForm.nrodoc < MINIMOS.MIN_NRODOC || this.postulanteForm.nrodoc > MAXIMOS.MAX_NRODOC) {
      this.dniErrorFlag = true;

    }
    // Valida que el numero fiscal contenta como minimo 11 cifras
    if (this.postulanteForm.nrofiscal < MINIMOS.MIN_NROFISCAL || this.postulanteForm.nrofiscal > MAXIMOS.MAX_NROFISCAL) {
      this.cuilErrorFlag = true;
    }

    // Valida que los valores de los selects autocompletar matchen con algun item de su collecion
    this.selectAutoErrorFlag = this.helperValidar.validateAutocompletes(this.childComponents);

    if (this.dniErrorFlag || this.cuilErrorFlag || this.selectAutoErrorFlag) {
      return false;
    }
    return true;
  }

  /* ---------------------------------------------- CALCULAR FECHA ----------------------------------------------------  */

  /**
   * Retorna fecha actual restando cantidad de años que se le pasa por parametro
   * ej: hoy es 21/03/2024 y se le restan 18, la fecha es 21/03/2006 (misma fecha 18 años atras)
   */
  private calculateDate(anios: number): string {
    //obtiene fecha actual
    const fechaActual = new Date();

    // Obtiene el año resntando la cantidad de años enviados por parametro
    const anio = fechaActual.getFullYear() - anios;
    // Obtiene mes y día actual de forma individual y los parsea a String
    const mes = String(fechaActual.getMonth() + 1).padStart(2, '0');
    const dia = String(fechaActual.getDate()).padStart(2, '0');

    //retorna fecha en forma de string
    return `${anio}-${mes}-${dia}`;
  }

  /* ---------------------------------------------- GET FILE Functions -----------------------------------------  */
  getFile(event: any): void {
    const fileInput = event.target;
    const file: File = fileInput.files[0];

    if (file) {
      this.codeandoToBase64 = true;

      const fileName = file.name;

      this.recortarImagenRedondaCentrada(file).then((fotoData: string) => {
        this.postulanteForm.foto.data = fotoData;
        this.postulanteForm.foto.extension = fileName.slice((fileName.lastIndexOf('.') - 1 >>> 0) + 2);
        this.postulanteForm.foto.name = fileName.slice(0, fileName.lastIndexOf('.'));
        this.postulanteForm.foto.path = "";

        this.updatePreviewPicture();

        this.codeandoToBase64 = false;

      }).catch((error: any) => {
        console.error('Error al recortar la imagen:', error);
        this.codeandoToBase64 = false;
      });
    }
  }

  async recortarImagenRedondaCentrada(file: File): Promise<string> {
    return new Promise<string>((resolve, reject) => {
      const reader = new FileReader();

      reader.onload = async () => {
        const img = new Image();
        img.src = reader.result as string;

        img.onload = () => {
          const canvas = document.createElement('canvas');
          const ctx = canvas.getContext('2d');

          let size = Math.min(img.width, img.height);

          canvas.width = size;
          canvas.height = size;

          // Determinar el punto de inicio para el recorte
          let startX = 0;
          let startY = 0;
          if (img.width > img.height) {
            startX = (img.width - img.height) / 2;
          } else {
            startY = (img.height - img.width) / 2;
          }

          // Dibujar un círculo
          ctx?.beginPath();
          ctx?.arc(size / 2, size / 2, size / 2, 0, Math.PI * 2);
          ctx?.clip();

          // Dibujar la imagen en el lienzo
          ctx?.drawImage(img, startX, startY, size, size, 0, 0, size, size);
          ctx?.closePath();

          // Convertir el canvas a una imagen base64
          resolve(canvas.toDataURL('image/png'));
        };

        img.onerror = () => {
          reject('No se pudo cargar la imagen.');
        };
      };

      reader.onerror = () => {
        reject('Error al leer la imagen.');
      };

      reader.readAsDataURL(file);
    });
  }

  //convierte el file obtenido en a base 64
  async convertFileToBase64(file: File): Promise<string> {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = () => {
        resolve(reader.result as string);
      };
      reader.onerror = (error) => {
        reject(error);
      };
      // Leer el archivo como ArrayBuffer
      reader.readAsDataURL(file); 
    });
  }


  updatePreviewPicture() {
    // si tiene contenido en el path, imagen que viene del backend
    if (this.postulanteForm.foto.path != null && this.postulanteForm.foto.path != "") {
      const urlImage = `${API_BASE}${this.postulanteForm.foto.path}?${new Date().getTime()}`;
      this.previewImage = urlImage;
      // si tiene una foto encodeada en base64, imagen cargada en el fron
    } else if (this.postulanteForm.foto.data != null && this.postulanteForm.foto.data != "") {

      this.previewImage = this.postulanteForm.foto.data
    }
  }

  /* ---------------------------------------------- ON DESTROY ----------------------------------------------------  */

  onDestroy(): void {
    this.perfilPostulanteSuscription.unsubscribe();
  }

}
