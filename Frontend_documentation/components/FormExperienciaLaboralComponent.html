import { Component, OnInit } from '@angular/core';
import { ExperienciaLaboral } from 'src/app/interfaces/ExperienciaLaboral';
import { Observable, Subscription } from 'rxjs';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';

@Component({
  selector: 'app-form-experiencia-laboral',
  templateUrl: './form-experiencia-laboral.component.html',
  styleUrls: ['./form-experiencia-laboral.component.css']
})

/**
 * Componente que contiene el formulario de carga de experiencia laboral. 
 * Contiene ademas un array con las experiencias laborales que se van cargando.
 */
export class FormExperienciaLaboralComponent implements OnInit {

  // si existen experiencias laborales precargados son recibidos por aqui
  experienciaLaboral: ExperienciaLaboral[] = [];

  // flag utilizado para la carga de datos en caso de edición
  editandoFlag: boolean = false;

  // item sobre el que se está operando, en caso de alta, modificación y baja
  experienciaLaboralActual: ExperienciaLaboral = {
    puesto: "",
    funcion: "",
    empresa: "",
    fecingreso: "",
    fecegreso: "",
    refApenom: "",
    telefono: NaN
  }

  // almacena temporalmente el item original que debe ser editado una vez completado el formulario de edición
  itemParaEditar!: ExperienciaLaboral | undefined;

  // fecha actual
  fechaHoy: String;

  /**
   * Datos CV Servicio
   */
  datosCVObservable$!: Observable<DatosAdicionales>;
  datosCVSuscription!: Subscription;
  datosCVForm!: DatosAdicionales;

  constructor(private datosCVService: DatosAdicionalesService) {
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = JSON.parse(JSON.stringify(datosCVFormService));
      this.experienciaLaboral = JSON.parse(JSON.stringify(this.datosCVForm.experienciaLaboral));
    });
    const hoyDate = new Date()
    this.fechaHoy =
      `${hoyDate.getFullYear()}-${hoyDate.getMonth() + 1 < 10 ? '0' : ''}${hoyDate.getMonth() + 1}-${hoyDate.getDate() < 10 ? '0' : ''}${hoyDate.getDate()}`
  }

  async ngOnInit(): Promise<void> {
  }

  /**
  * Actualiza el observable y avisa al postulanteService del cambio
  */
  updateDatosCVLocal() {
    this.datosCVForm.experienciaLaboral = this.experienciaLaboral;
    this.datosCVService.updateEstadoDatosCVLocal(this.datosCVForm);
  }

  /**
   * TODO: Cuando el componente se separe, esta debera actualizar la info de su hijo(FormNivelEducativo). Y si todo
   * sale bien reien ahi navegar hacia cursos
   */
  actualizarInfo(): boolean {
    // Se actualizan los datos del cv del postulante
    this.updateDatosCVLocal();
    return true;
  }

  /**
   * Filtra el arreglo local de "experienciaLaboral", y actualiza el arreglo local 
   */
  eliminar() {
    console.log(this.experienciaLaboralActual);

    this.experienciaLaboral = this.experienciaLaboral.filter((item) =>
      item.puesto.toLowerCase() !== this.experienciaLaboralActual.puesto.toLowerCase() ||
      item.funcion.toLowerCase() !== this.experienciaLaboralActual.funcion.toLowerCase() ||
      item.empresa.toLowerCase() !== this.experienciaLaboralActual.empresa.toLowerCase() ||
      item.fecingreso !== this.experienciaLaboralActual.fecingreso
    )
    console.log(this.experienciaLaboral);

  }

  /**
   * Blanquea el formulario enlazado al experienciaLaboralActual y setea el editar como falso
   */
  abrirAgregar(): void {
    this.blanquearForm();
    this.editandoFlag = false;
  }

  /**
   * Si pasan las validaciones, clonaremos la informacion cargada del usuario y la agregamos al arreglo local
   * "experienciaLaboralActual", Por ultimo se blanquea el formulario.
   */
  agregarExperiencia() {

    if (!this.validar()) {
      return
    }

    // crea nuevo objeto para pushear al array de experiencias laborales
    const nuevaExperienciaLaboral = {
      puesto: this.experienciaLaboralActual.puesto.toLowerCase(),
      funcion: this.experienciaLaboralActual.funcion.toLowerCase(),
      empresa: this.experienciaLaboralActual.empresa.toLowerCase(),
      fecingreso: this.experienciaLaboralActual.fecingreso,
      fecegreso: this.experienciaLaboralActual.fecegreso,
      refApenom: this.experienciaLaboralActual.refApenom.toLowerCase(),
      telefono: this.experienciaLaboralActual.telefono
    }

    // Agregamos el nuevo objeto al arreglo local
    this.experienciaLaboral.push(nuevaExperienciaLaboral);

    console.log(this.experienciaLaboral);

    // Blanqueamos el formulario
    this.blanquearForm();
  }

  /**
   * Activa la bandera de edicion y actualiza el formulario con la info de la card (param item)
   * @param item EducacionPostulante
   */
  abrirEditar(item: ExperienciaLaboral): void {
    this.editandoFlag = true;
    // Buscamos el objeto enlazado a la card y lo guardamos para no perder referencia al momento de actualizarlo
    this.itemParaEditar = this.experienciaLaboral.find(itemArray =>
      itemArray.puesto === item.puesto
      && itemArray.empresa === item.empresa
      && itemArray.fecingreso === item.fecingreso
    )
    this.actualizarExperienciaLaboral(item);
  }

  /**
   * Si pasan las validaciones, buscamos el objeto enlazado a la card en el arreglo "educacionPostulante"
   * Si lo encontramos actualizamos la info de la card con la nueva informacion del formulario. Por ultimo
   * cerramos la bandera de edicion y blanqueamos el formulario.
   */
  editarExperiencia(): void {

    if (!this.validar()) {
      return
    }

    if (this.itemParaEditar === undefined) {
      return
    }

    // guardamos la referencia del objeto a editar para luego poder sobreescribirlo
    this.itemParaEditar.puesto = this.experienciaLaboralActual.puesto,
      this.itemParaEditar.funcion = this.experienciaLaboralActual.funcion,
      this.itemParaEditar.empresa = this.experienciaLaboralActual.empresa,
      this.itemParaEditar.fecingreso = this.experienciaLaboralActual.fecingreso
    this.itemParaEditar.fecegreso = this.experienciaLaboralActual.fecegreso
    this.itemParaEditar.refApenom = this.experienciaLaboralActual.refApenom
    this.itemParaEditar.telefono = this.experienciaLaboralActual.telefono

    // Cerramos la bandera de edicion
    this.editandoFlag = false;
    this.blanquearForm();
  }

  /**
   * Valida el formulario de experiencia laboral tanto por bootstrap como validaciones manuales
   * @returns boolean
   */
  validar(): boolean {
    const form: HTMLFormElement | null = document.querySelector("#formExperienciaLaboral");
    let esValido = true;

    if (form == null) {
      return false
    }

    //Validamos por Bootstrap y checkeamos las validaciones anteriores
    if (form.checkValidity() && esValido) {
      //Si paso y antes se habia validado
      if (form.classList.contains("was-validated")) {
        //Normalizamos el formulario
        form.classList.remove("was-validated")
      }
      //forzamos el cierre del form si todo salio bien
      this.cerrarForm();
      return true;
    }

    //Si no paso validamos la interfas
    form.classList.add("was-validated");
    return false;
  }

  /**
   * Actualiza el objeto enlazado al formulario con informacion vacia
   */
  blanquearForm() {
    this.experienciaLaboralActual = {
      puesto: "",
      funcion: "",
      empresa: "",
      fecingreso: "",
      fecegreso: "",
      refApenom: "",
      telefono: NaN
    };
  }

  /**
   * Actualiza el objeto enlazado al formulario con la informacion que venga por parametros
   * @param item EducacionPostulante
   */
  actualizarExperienciaLaboral(item: ExperienciaLaboral) {
    // Acutaliza los datos de experienciaLaboralActual 
    this.experienciaLaboralActual = {
      puesto: item.puesto.toLowerCase(),
      funcion: item.funcion.toLowerCase(),
      empresa: item.empresa.toLowerCase(),
      fecingreso: item.fecingreso,
      fecegreso: item.fecegreso,
      refApenom: item.refApenom.toLowerCase(),
      telefono: item.telefono
    };
  }

  /**
   *  Cierre forzado del formulario modal
   */
  cerrarForm(): void {
    const buttonCerrar: HTMLButtonElement | null = document.querySelector("#btn-cerrar-form");
    this.normalizarForm();

    if (buttonCerrar == null) {
      return
    }
    buttonCerrar.click();
  }

  /**
   * Elimina los estilos de error de boostrap si el form esta vacío
   */
  private normalizarForm() {
    const form: HTMLFormElement | null = document.querySelector("#formExperienciaLaboral");

    if (form != null) {
      if (form.classList.contains("was-validated")) {
        form.classList.remove("was-validated");
      }
    }
  }
}
