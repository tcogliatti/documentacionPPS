import { Component, Input, OnInit } from '@angular/core';
import { MINIMOS } from '../../../constants'
import { ConocimientoInformatica } from 'src/app/interfaces/ConocimientoInformatica';
import { Observable, Subscription } from 'rxjs';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';

@Component({
  selector: 'app-form-informatica',
  templateUrl: './form-informatica.component.html',
  styleUrls: ['./form-informatica.component.css']
})

/**
 * Componente que contiene el formulario de los conocimientos informáticos, y maneja los datos cargados de informática.
 */
export class FormInformaticaComponent implements OnInit {
  // si existen conocimeintos informáticos precargados son recibidos por aqui
  conocimientosEnInformatica: ConocimientoInformatica[] = [];

  //array de años para carga
  years: number[] = [];

  // flag utilizado para la carga de datos en caso de edición
  editandoFlag: boolean = false;

  // item sobre el que se está operando, en caso de alta, modificación y baja
  conocimientoInformaticaActual: ConocimientoInformatica = {
    idinformatica: NaN,
    idpostulante: NaN,
    descripcion: "",
    anio: "",
    estado: "",
    nivel: ""
  }

  // almacena temporalmente el item original que debe ser editado una vez completado el formulario de edición
  itemParaEditar!: ConocimientoInformatica | undefined;

  /**
   * Datos CV Servicio
   */
  datosCVObservable$!: Observable<DatosAdicionales>;
  datosCVSuscription!: Subscription;
  datosCVForm!: DatosAdicionales;

  constructor(private datosCVService: DatosAdicionalesService) {
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = JSON.parse(JSON.stringify(datosCVFormService));
      this.conocimientosEnInformatica = JSON.parse(JSON.stringify(this.datosCVForm.conocimientosInformatica));
    });
  }

  async ngOnInit(): Promise<void> {
    this.createYeatsSelect();
  }

  /**
   * Actualiza el observable y avisa al postulanteService del cambio
   */
  updateDatosCVLocal() {
    this.datosCVForm.conocimientosInformatica = this.conocimientosEnInformatica;
    this.datosCVService.updateEstadoDatosCVLocal(this.datosCVForm);
  }

  /**
   * TODO: Cuando el componente se separe, esta debera actualizar la info de su hijo(FormNivelEducativo). Y si todo
   * sale bien reien ahi navegar hacia cursos
   */
  actualizarInfo(): boolean {
    // Se actualizan los datos del cv del postulante
    this.updateDatosCVLocal();
    return true;
  }
  posInfo(): boolean {
    this.updateDatosCVLocal();
    this.datosCVService.persistDatosCV();
    return true;
  }

  /**
   * Filtra el arreglo local de "conocimientosEnInformatica", y actualiza el arreglo local 
   */
  eliminar() {
    this.conocimientosEnInformatica = this.conocimientosEnInformatica.filter(item =>
      item.descripcion.toLowerCase() != this.conocimientoInformaticaActual.descripcion.toLowerCase()
      || item.nivel.toLowerCase() != this.conocimientoInformaticaActual.nivel.toLowerCase()
      || item.estado.toLowerCase() != this.conocimientoInformaticaActual.estado.toLowerCase()
      || item.anio != this.conocimientoInformaticaActual.anio
    )
  }
  /**
   * Blanquea el formulario enlazado al conocimientoInformaticaActual y setea el editar como falso
   */
  abrirAgregar(): void {
    this.blanquearForm();
    this.editandoFlag = false;
  }

  /**
   * Si pasan las validaciones, clonaremos la informacion cargada del usuario y la agregamos al arreglo local
   * "conocimientoInformaticaActual", Por ultimo se blanquea el formulario.
   */
  agregarInformatica() {

    if (!this.validar()) {
      return
    }

    const nuevoConocimientoInformatico: ConocimientoInformatica = {
      idinformatica: this.conocimientoInformaticaActual.idinformatica,
      idpostulante: this.conocimientoInformaticaActual.idpostulante,
      descripcion: this.conocimientoInformaticaActual.descripcion.toLowerCase(),
      anio: this.conocimientoInformaticaActual.anio,
      estado: this.conocimientoInformaticaActual.estado.toLowerCase(),
      nivel: this.conocimientoInformaticaActual.nivel.toLowerCase()
    }

    // Agregamos el nuevo objeto al arreglo local
    this.conocimientosEnInformatica.push(nuevoConocimientoInformatico);

    // Blanqueamos el formulario
    this.blanquearForm();
  }

  /**
   * Activa la bandera de edicion y actualiza el formulario con la info de la card (param item)
   * @param item EducacionPostulante
   */
  abrirEditar(item: ConocimientoInformatica): void {
    this.editandoFlag = true;
    // Buscamos el objeto enlazado a la card y lo guardamos para no perder referencia al momento de actualizarlo
    this.itemParaEditar = this.conocimientosEnInformatica.find(itemArray =>
      itemArray.descripcion.toLowerCase() === item.descripcion.toLowerCase()
      && itemArray.nivel.toLowerCase() === item.nivel.toLowerCase()
      && itemArray.estado.toLowerCase() === item.estado.toLowerCase()
      && itemArray.anio === item.anio
    )
    this.actualizarConocimientoInformaticoLocal(item);
  }

  /**
   * Si pasan las validaciones, buscamos el objeto enlazado a la card en el arreglo "educacionPostulante"
   * Si lo encontramos actualizamos la info de la card con la nueva informacion del formulario. Por ultimo
   * cerramos la bandera de edicion y blanqueamos el formulario.
   */
  editarInformatica(): void {

    if (!this.validar()) {
      return
    }

    if (this.itemParaEditar === undefined) {
      return
    }

    this.itemParaEditar.descripcion = this.conocimientoInformaticaActual.descripcion.toLowerCase()
    this.itemParaEditar.anio = this.conocimientoInformaticaActual.anio
    this.itemParaEditar.estado = this.conocimientoInformaticaActual.estado.toLowerCase()
    this.itemParaEditar.nivel = this.conocimientoInformaticaActual.nivel.toLowerCase()

    // Cerramos la bandera de edicion
    this.editandoFlag = false;
    this.blanquearForm();
  }

  /**
   * Valida el formulario de conocimientos informaticos tanto por bootstrap como validaciones manuales
   * @returns boolean
   */
  validar(): boolean {
    const form: HTMLFormElement | null = document.querySelector("#formConocimientoInformatica");
    let esValido = true;

    if (form == null) {
      return false
    }

    //Validamos que el estado sea correcto
    let estado = this.conocimientoInformaticaActual.estado.toLowerCase();
    if (estado != "incompleto" && estado != "en curso" && estado != "completo") {
      esValido = false;
      this.conocimientoInformaticaActual.estado = ""
    }

    //Validamos que el nivel sea correcto
    let nivel = this.conocimientoInformaticaActual.nivel.toLowerCase();
    if (nivel != "basico" && nivel != "intermedio" && nivel != "avanzado") {
      esValido = false;
      this.conocimientoInformaticaActual.nivel = ""
    }

    //Validamos por Bootstrap y checkeamos las validaciones anteriores
    if (form.checkValidity() && esValido) {
      //Si paso y antes se habia validado
      if (form.classList.contains("was-validated")) {
        //Normalizamos el formulario
        form.classList.remove("was-validated")
      }
      //forzamos el cierre del form si todo salio bien
      this.cerrarForm();
      return true;
    }

    //Si no paso validamos la interfas
    form.classList.add("was-validated");
    return false;
  }

  /**
   * Actualiza el objeto enlazado al formulario con informacion vacia
   */
  blanquearForm() {
    this.conocimientoInformaticaActual = {
      idinformatica: NaN,
      idpostulante: NaN,
      descripcion: "",
      anio: "",
      estado: "",
      nivel: ""
    };
  }

  /**
   * Actualiza el objeto enlazado al formulario con la informacion que venga por parametros
   * @param item EducacionPostulante
   */
  actualizarConocimientoInformaticoLocal(item: ConocimientoInformatica) {
    // Acutaliza los datos de conocimientoInformaticaActual 
    this.conocimientoInformaticaActual = {
      idinformatica: item.idinformatica,
      idpostulante: item.idpostulante,
      descripcion: item.descripcion.toLowerCase(),
      anio: item.anio,
      estado: item.estado.toLowerCase(),
      nivel: item.nivel.toLowerCase()
    };
  }

  /**
   *  Cierre forzado del formulario modal
   */
  cerrarForm(): void {
    const buttonCerrar: HTMLButtonElement | null = document.querySelector("#btn-cerrar-form");
    this.normalizarForm()

    if (buttonCerrar == null) {
      return
    }
    buttonCerrar.click();
  }

  /**
   * Crea una lista de años desde el año actual hasta MINIMOS.ANIOS_SELECT
   * para el select año
   */
  createYeatsSelect() {
    const esteAnio = new Date().getFullYear();
    for (let index = 0; index < MINIMOS.ANIOS_SELECT; index++)
      this.years.push(esteAnio - index)
  }

  /**
   * Elimina los estilos de error de boostrap si el form esta vacío
   */
  private normalizarForm() {
    const form: HTMLFormElement | null = document.querySelector("#formConocimientoInformatica");

    if (form != null) {
      if (form.classList.contains("was-validated")) {
        form.classList.remove("was-validated");
      }
    }
  }
}
