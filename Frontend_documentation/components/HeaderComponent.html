import { Component, OnInit } from '@angular/core';
import { NavigationEnd, Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
// import { PostulanteFakeHeader } from 'src/app/interfaces/PostulanteFakeHeder';
import { PostulantePerfil } from 'src/app/interfaces/PostulantePerfil';
import { PostulanteService } from 'src/app/servicios/postulante.service';
import { ROUTES, API_BASE } from 'src/constants';
import { DatosAdicionalesService } from '../../servicios/datos-adicionales.service';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.css']
})
export class HeaderComponent implements OnInit {

  existPostulante$: Observable<boolean>;
  existPostulanteSuscripcion: Subscription;

  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteForm!: PostulantePerfil;

  previewImage: string | null = null;

  /**
   * routes: variable instanciada a partir de la constante ROUTES 
   */
  routes: any;

  /**
   * isNew: varibale que renderiza condicionalmente el navbar (Carga de datos o Resumen)
   */
  isNew!: boolean;

  /**
   * routerSubcsiption: variable encargada de escuchar el evento: NavigationEnd
   */
  routerSubscription: Subscription | undefined;

  /**
   * fullPath: variable local que verificara la url y
   * activara los links del navBar en su debido caso
   */
  fullPath!: string;

  /* DATOS CV SERVICE */
  datosCVObservable$: Observable<DatosAdicionales>;
  datosCVSuscription: Subscription;
  datosCVForm!: DatosAdicionales;
  noExistDatosCV$: Observable<boolean>;

  constructor(private router: Router, private postulanteService: PostulanteService, private datosCVService: DatosAdicionalesService) {
    this.existPostulante$ = this.postulanteService.tienePerfil.asObservable();
    this.existPostulanteSuscripcion = this.existPostulante$.subscribe(datoTienePerfil => {
      this.isNew = !datoTienePerfil;
    })
    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable();
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(perfilFromService => {
      this.postulanteForm = { ...perfilFromService }
      // verifica que la foto de perfil exista, y le aplica la ruta completa
      if (this.postulanteForm.foto.path || this.postulanteForm.foto.path != "")
        this.previewImage = API_BASE + this.postulanteForm.foto.path + '?' + new Date().getTime();
      else
        this.previewImage = null
    })
    /* DATOS CV SERVICE */
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = { ...datosCVFormService }
    });
    this.noExistDatosCV$ = this.datosCVService.noTieneDatosCV.asObservable();
  }

  /**
   * OnInit
   * 1. Inicializa la variable local routes con la info del ROUTES.
   * 2. inicializamos la variable isNew en fasle o true.
   * 3. Obtiene una actualización del ultimo path en el que se encuetra actualmente y actualiza el estado del componente.Es para el caso exepcional en que el usuario ingrese diorectamente al flow de carga introduciendo una URL en el navegador, la cual no corresponda al paso 1 del flow.
   * 4. Suscribe al evento de NavigationEnd para obtener la URL del navegador. Si detecta un cambio en la URL, entonces ejecuta la actualización del componente.
   */
  ngOnInit(): void {
    // 1
    this.routes = ROUTES;

    // 2 
    this.isNew = true;

    // 3
    const path = this.router.url;
    this.fullPath = path;

    // 4
    this.routerSubscription = this.router.events.subscribe(event => {
      if (event instanceof NavigationEnd) {
        this.fullPath = event.url;
      }
    });

  }


  /**
  * ngOnDestroy
  * se des-suscribe de la dependencia que le permite obtener la URL
  */
  ngOnDestroy(): void {
    if (this.routerSubscription) {
      this.routerSubscription.unsubscribe();
    }
    this.existPostulanteSuscripcion.unsubscribe();
    this.perfilPostulanteSuscription.unsubscribe();
  }
}
