import { Component, EventEmitter, OnInit, Output, enableProdMode } from '@angular/core';
// import { StepProgressBar } from 'src/app/interfaces/StepProgressBar';
import { StepProgressBar } from '../../interfaces/StepProgressBar';
import { ROUTES } from '../../../constants';
@Component({
  selector: 'app-layout-flow-form-cv',
  templateUrl: './layout-flow-form-cv.component.html',
  styleUrls: ['./layout-flow-form-cv.component.css']
})

/**
 * Layout para el flow de carga de datos adicionales de CV
 * 
 * Este layout es necesario para:
 *  1. enviar la configurar a la barra de progreso
 *  2. mantener el flujo de carga como una SPA, para que la barra de progreso no se refresque cada vez que se pasa de un form a otro
 * 
 * Define dos variables claves para trasnmitir y configurar la barra de progreso
 *  - progressBarInit @type StepProgressBar[]: se trata de un arreglo de objetos definidos por la interface StepProgressBar. Proporciona la cantidad de pasos, la URL de cada uno y su Titulo, necesarios para el correcto funcionamiento de la barra de progreso.
 *  - titleLayout @type string: titulo del flow de carga que debe mostrarse en la barra de progreso
 */
export class LayoutFlowFormCvComponent implements OnInit {

  progressBarInit: StepProgressBar[];
  titleLayout: string;
  /**
   * evento que se emite desde el layout para que la barra de progreso se configure correctamente
   */
  @Output()
  emitProgressBarInit: EventEmitter<StepProgressBar[]> = new EventEmitter<StepProgressBar[]>();
  /**
   * Evento que se emite para que la barra de progreso reciba el titulo del flow de carga
   */
  @Output()
  emitTitleLayout: EventEmitter<string> = new EventEmitter<string>();

  /**
   * Constructor
   * Se inicializan las variables de configuraci贸n de la barra de progreso
   */
  constructor() {
    this.titleLayout = "Completar CV";

    this.progressBarInit = [
      {
        path: ROUTES.NIVEL_EDUCATIVO,
        name: "Educacion",
      },
      {
        path: ROUTES.CURSO,
        name: "Curso",
      },
      {
        path: ROUTES.EXPERIENCIA_LABORAL,
        name: "Experiencia Laboral",
      },
      {
        path: ROUTES.IDIOMA,
        name: "Idioma",
      },
      {
        path: ROUTES.INFORMATICA,
        name: "Informatica",
      },
    ]
  }
  /**
   * ngOnInit
   * Se emiten las variables de configuraci贸n de la barra de progreso
   */
  ngOnInit(): void {
    this.emitProgressBarInit.emit(this.progressBarInit);
    this.emitTitleLayout.emit(this.titleLayout);
  }
}
