import { Component, Input, OnInit } from '@angular/core';
import { StepProgressBar } from '../../interfaces/StepProgressBar';
import { NavigationEnd, Router } from '@angular/router';
import { Subscription } from 'rxjs';

@Component({
  selector: 'app-progress-bar-form',
  templateUrl: './progress-bar-form.component.html',
  styleUrls: ['./progress-bar-form.component.css']
})

/**
 * Barra de progreso de carga de datos
 * 
 * Descripción:
 * Muestra a los usuarios un diseño que le informa el progreso de la carga de datos. 
 * Es utilizado en flujos en que se requiere completar información en varios pasos, por ejemplo, donde cada paso es un formulario.
 * Para informar al usuario el estado de la carga, el diseño UX/UI del componente debe mostrar:
 *  1. Una serie de items que representen cada paso del flujo
 *  2. Cuál es el paso actual en el que se encuentra
 *  3. Cuáles son los pasos por completarse
 *  4. Cuáles son los pasos que ya se completaron 
 * 
 * Configuración:
 * Este componente debe montarse en un layout. El layout contendrá esta barra de progreso así como también las diferentes páginas que consisten en cada paso de la carga de datos.
 * Es importante que esas diferentes páginas contenidas en el layout estén previamente definidas en el app-routing.module.ts con sus respectivos PATH.
 * El layout proveerá al componeten ProgressBarForm la configuración con la que debe inicializarse. Esto brinda flexibilidad al componente pudiendo ser reutilizado en diferentes contextos que le enviarán diferentes configuraciones.
 * 
 * El componente ProgressBarForm recibirá la configuracíon desde el layout que lo contiene.
 *  @input progressBarInit<StepProgressBar>[]
 * Este array define: 
 *  - length: cantidad de pasos.
 *  - index de cada elemento: define su orden
 *  - Cada elemento: contiene un objeto de tipo StepProgressBar que define el de cada paso, su PATH y su nombre:
 *    a. Titulo del paso
 *    b. Path del formulario. Es importante que este valor sea exatamente igual al PATH definido en app-routing.module.ts, correspondiente a cada children del layout.
 * 
 * La barra de progreso también recibe del layout el nombre generál del flujo de carga de datos:
 * @input nombreDeFlujo: string
 * 
 * Funcionamiento:
 *  1. El componente lee la URL en la que se encuentra
 *  2. Obtiene el último tramo del path
 *     Ejemplo 
 *      URL completa: http://localhost:4200/datos-requeridos/datos-personales
 *      ultimo tramo de la URL: "datos-personales"
 *  3. Busca en el array de configuración el path que coincide con el de la URL
 *  4. obtiene el index y lo asigna a una variable llamada actualStep: number
 *      actualStep = index + 1
 *  5. Obtiene el nombre del paso actual
 *  6. Con estos datos el componente reconoce el estado en que se ecuentra cada ítem del flujo de carga
 *     a. paso actual: actualStep
 *     b. Pasos completados: todos los index < actualStep - 1
 *     c. Pasos por completarse: todos los index > actualStep - 1
 *  7. Renderido condicional aplicando clases diferenciales a cada ítem del flujo de carga de acuerdo al estado que le corresponde
 *
 */
export class ProgressBarFormComponent implements OnInit {

  /**
   * Array de configuración
   * @type {StepProgressBar[]}
   */
  @Input()
  progressBarInit: StepProgressBar[] = [];

  /**
   * Titulo del flow de carga
   * @type {string}
   */
  @Input()
  titleLayout: string = ""

  routerSubscription: Subscription | undefined;
  actualStep: number = NaN;
  steps: number = NaN;

  /**
   * Constructor
   * @param router Iyección de dependencia de Router para trabajar con la URL del navegador
   */
  constructor( private router: Router ) {
  }

  /**
   * OnInit
   * 1. inicializa la variable Steps: total de pasos
   * 2. Obtiene una actualización del path en el que se encuetra actualmente y actualiza el estado del componente.Es para el caso exepcional en que el usuario ingrese diorectamente al flow de carga introduciendo una URL en el navegador, la cual no corresponda al paso 1 del flow.
   * 3. Suscribe al evento de NavigationEnd para obtener la URL del navegador. Si detecta un cambio en la URL, entonces ejecuta la actualización del componente.
   */
  ngOnInit(): void {
    // 1. obtener la cantidad de pasos
    this.steps = this.progressBarInit.length;

    // 2. verifica en que tramo del flow se encuentra al iniciar el componente
    const path = this.router.url
    this.updateStep(path)
    
   
    // 3. Suscribirse al evento NavigationEnd para detectar cambios en la navegación
    this.routerSubscription = this.router.events.subscribe(event => {
      if (event instanceof NavigationEnd) {
        this.updateStep(event.url)

      }
    });
  }
 /**
  * updateStep
  *   1. obtiene el ultimo segment de la URL
  *   2. Busca en el array a que paso corresponde ese segmento. Toma el index de ese paso, actualiza la variable actualStep, utilizada para el renderizado condicional.
  * @param url la URL completa obtenida del navegador
  */
  updateStep(url: string): void {
    // 1. obtiene la url y se queda con el ultimo segmento de la ruta
    const urlSinParams = url.split('?')[0];
    const partesUrl = urlSinParams.split('/');
    const path = partesUrl[partesUrl.length - 1];
        // 2. obtiene el numero de paso al corresponde ese path
    const index = this.progressBarInit.findIndex(item => item.path === path);
    if(index !== -1)
      this.actualStep = index + 1
    else
      this.actualStep = NaN
  }

  /**
   * ngOnDestroy
   * se des-suscribe de la dependencia que le permite obtener la URL
   */
  ngOnDestroy(): void {
    if (this.routerSubscription) {
      this.routerSubscription.unsubscribe();
    }
  }
}
<div class="d-flex flex-column flex-md-row progress-bar-box justify-content-center fondo-nav">
    <!-- nombre de flow de carga -->
    <div *ngIf="titleLayout" class="d-none d-sm-flex flex-sm-column justify-content-center col-md-2">
        <h4 class="text-center fw-normal mb-sm-0 mb-md-2" data-qa="nameFlow">{{titleLayout}}</h4>
    </div>
    <!-- nombre del paso actual -->

    <div *ngIf="progressBarInit.length > 0" class="mb-sm-1 mb-md-0 col-md-3 d-flex flex-column justify-content-center">
        <h5 class="text-center text-md-end fw-normal" data-qa="nameStep">{{progressBarInit[actualStep-1].name}}</h5>
    </div>
    <!-- linea de separacion (no es visible en mobile) -->
    <div *ngIf="progressBarInit.length > 0" class="d-none d-md-flex flex-column justify-content-center col-1">
        <div class="line-separation mx-3 mb-1"></div>
    </div>
    <!-- renderizado de sección indicadora de los pasos -->
    <div *ngIf="progressBarInit.length > 0"
        class="p-0 mb-1 d-flex justify-content-center justify-content-md-start gap-2 gap-md-8 dots-box col-md-6">
        <div class="d-flex mb-sm-1 mb-md-0 mb-md-0">
            <div class="d-flex align-content-center" *ngFor="let step of progressBarInit; let i = index">
                <!-- puntos de cada paso -->
                <div class="d-flex flex-column justify-content-center">
                    <div [ngClass]="{
                            'progress-bar-item' : true,  
                            'active' : actualStep==i+1,  
                            'complete' : actualStep>i+1 
                            }">
                        <!-- puntito que va en el centro del item activo -->
                        <div class="inisde-dot twinkle-animation" *ngIf="actualStep==i+1"></div>
                    </div>
                </div>
                <!-- linea que une los puntos entre si -->
                <div class="d-flex flex-column justify-content-center">
                    <div [ngClass]="{'line-flow': true, 'complete' : actualStep>i+1 }" *ngIf="i+1!=steps"></div>
                </div>
            </div>
        </div>
        <!-- contador de pasos -->
        <div class="d-flex flex-column justify-content-center">
            <p class="mb-1 mb-sm-2 mb-md-1 ms-md-1 f-6">paso {{actualStep}}/{{steps}}</p>
        </div>
    </div>
    <div *ngIf="progressBarInit.length == 0" class="d-flex flex-column justify-content-center">
        <h4 class="text-center fw-normal">😞 No existen pasos para mostrar...</h4>
    </div>
</div>
