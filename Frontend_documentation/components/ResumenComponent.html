import { Component, OnInit } from '@angular/core';
import { PostulanteService } from "../../servicios/postulante.service";
import { Observable, Subscription } from "rxjs";
import { PostulantePerfil } from "../../interfaces/PostulantePerfil";
import { API_BASE, ROUTES } from 'src/constants';
import { animate, style, transition, trigger } from '@angular/animations';
import { DatosAdicionalesService } from 'src/app/servicios/datos-adicionales.service';
import { DatosAdicionales } from 'src/app/interfaces/DatosAdicionales';
import { ConocimientoInformatica } from 'src/app/interfaces/ConocimientoInformatica';

@Component({
  selector: 'app-resumen',
  templateUrl: './resumen.component.html',
  styleUrls: ['./resumen.component.css'],
  animations: [
    trigger('fadeInLeft', [
      transition(':enter', [
        style({ opacity: 0, transform: 'translateX(-25%)' }),
        animate('1000ms ease-out', style({ opacity: 1, transform: 'translateX(0)' }))
      ])
    ])
  ]
})
export class ResumenComponent implements OnInit {
  routes: any = ROUTES;
  previewImage: string | null = null;

  /* POSTULANTE SERVICE */
  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteForm!: PostulantePerfil;
  existPostulante$: Observable<boolean>;
  nombreCvSubido!: string;

  /* DATOS CV SERVICE */
  datosCVObservable$: Observable<DatosAdicionales>;
  datosCVSuscription: Subscription;
  datosCVForm!: DatosAdicionales;

  noExistDatosCV$: Observable<boolean>;

  constructor(private postulanteService: PostulanteService, private datosCVService: DatosAdicionalesService) {
    /* POSTULANTE SERVICE */
    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable();
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(postulanteFromService => {
      this.postulanteForm = { ...postulanteFromService }

      // verifica que la foto de perfil exista, y le aplica la ruta completa
      if (this.postulanteForm.foto.path || this.postulanteForm.foto.path != "")
        this.previewImage = API_BASE + this.postulanteForm.foto.path + '?' + new Date().getTime();
      else
        this.previewImage = null

      if (this.postulanteForm.rutacv != null && this.postulanteForm.rutacv != "")
        this.nombreCvSubido = decodeURIComponent(this.postulanteForm.rutacv.substring(this.postulanteForm.rutacv.lastIndexOf('/') + 1));

    })
    this.existPostulante$ = this.postulanteService.tienePerfil.asObservable();
    /* DATOS CV SERVICE */
    this.datosCVObservable$ = this.datosCVService.datosCVLocal.asObservable();
    this.datosCVSuscription = this.datosCVObservable$.subscribe(datosCVFormService => {
      this.datosCVForm = { ...datosCVFormService }
    });
    this.noExistDatosCV$ = this.datosCVService.noTieneDatosCV.asObservable();
  }

  async ngOnInit(): Promise<void> {
    // console.log("ONINIT RESUMEN: ", this.postulanteForm);
    this.routes = ROUTES;
    this.datosCVService.obtenerDatosCVByIdPostulante();
    window.scroll(0, 0);
    console.log(this.datosCVForm)
  }
  ngAfterViewInit(): void {
    // console.log("AFTERVIEWINIT RESUMEN: ", this.postulanteForm)

  }

  ngOnDestroy(): void {
    // console.log("ONDESTROY RESUMEN: ", this.postulanteForm)
  }

  /**
  * Descarga del backend del archivo de CV subido 
  * @returns
  */
  async downloadCvFile(): Promise<void> {
    try {
      const cvFile = await this.postulanteService.getFileCv()
      if (cvFile == null)
        return
      const urlBlob = window.URL.createObjectURL(cvFile);
      const link = document.createElement('a');
      link.href = urlBlob;
      link.download = this.nombreCvSubido;
      link.click();
      window.URL.revokeObjectURL(urlBlob);
    } catch (error) {
      throw error
    }
  }
}
