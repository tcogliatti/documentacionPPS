import { Component, EventEmitter, Input, OnInit, Output, ViewChild, QueryList } from '@angular/core';
import { HttpClient } from "@angular/common/http";
import { ViewEncapsulation } from '@angular/core';

@Component({
  selector: 'app-select-autcomplete',
  templateUrl: './select-autocomplete.component.html',
  styleUrls: ['./select-autocomplete.component.css'],
  encapsulation: ViewEncapsulation.None // Desactiva el encapsulamiento de estilos
})

/**
 * Componente de autocompletado
 * Este componente recibe una array de datos para el autocompletado y el dato seleccionado
 */
export class SelectAutocompleteComponent implements OnInit {

  /**
   * Objeto que seleccionara el usuario o en su defecto lo que tenga seleccionado
   */
  @Input()
  dato: any;

  /**
   * Coleccion de objetos que representaran a las opciones del input
   */
  @Input()
  collection: any;

  /**
   * Evento que actualizara el "dato" seleccionado
   */
  @Output()
  datoChange: EventEmitter<any> = new EventEmitter<any>();

  /**
   * Acceso directo al input autocompletar
   */
  @ViewChild('auto')
  auto: any;

  /**
   * Palabra clave requerida por los objetos a renderizar, por la cual se ordenaran las opciones
   */
  public keyword = "descripcion";

  valorAnterior!: string;

  /**
   * Bandera que servira para validar o no los inputs
   */
  @Input()
  esRequerido: boolean = false;

  constructor(private http: HttpClient) {
  }

  ngOnInit(): void {


    let inputsContainers = document.querySelectorAll('.input-container');
    // Itera sobre cada elemento y aplicar los cambios
    inputsContainers.forEach((inputC) => {

      // Obtiene el input dentro de este contenedor
      let inputElement: HTMLInputElement | null = inputC.querySelector('input');
      if (inputElement != null) {
        // Agrega la clase "form-select" al input y la flechita visualmente
        inputElement.classList.add("form-select");
        inputElement.classList.add("form-control");

      }
    });
    this.requeridos();
  }

  /**
   * Se ejecuta cada vez que el contenido del componente ha sido verificado
   * por el mecanismo de detección de cambios de Angular; se llama después
   * del método ngAfterContentInit. Este también se invoca en cada ejecución
   * posterior de ngDoCheck y está relacionado principalmente con las 
   * inicializaciones del componente hijo.
   */
  ngAfterContentChecked(): void {
    this.datoChange.emit(this.dato);
    this.valorAnterior = this.dato;
  };

  clear(): void {
    this.auto.clear();
  }

  onItemSelected(item: any) {
    this.dato = item;
    this.datoChange.emit(this.dato);
  }

  deselectedItem() {
    this.dato = null;
    this.datoChange.emit(this.dato);
  }

  requeridos() {
    let selects = document.querySelectorAll(".select-requerido");

    selects.forEach(select => {
      let input: HTMLInputElement | null = select.querySelector('input');
      if (input) {
        input.required = true;
      }
    });
  }
}
