import { HttpClient } from '@angular/common/http';
import { Component, OnInit } from '@angular/core';
import { Router, RouterLink } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
import { PostulantePerfil } from 'src/app/interfaces/PostulantePerfil';
import { PostulanteService } from 'src/app/servicios/postulante.service';
import { API_BASE, API_PATH, ROUTES } from 'src/constants'

@Component({
  selector: 'app-subir-archivo-cv',
  templateUrl: './subir-archivo-cv.component.html',
  styleUrls: ['./subir-archivo-cv.component.css']
})

/**
 * Componente para subir un archivo de CV
 * Tiene la responsabilidad de altas y bajas de este dato
 */
export class SubirArchivoCvComponent implements OnInit {

  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteLocal!: PostulantePerfil;
  tieneArchivoCv: boolean = true;
  httpError: boolean = false;
  isLoading: boolean = false;
  routes: any = ROUTES;
  selectedFile: File | null = null;
  nombreCvSubido: string = '';
  cvFile: Blob | null = null;

  constructor(private postulanteService: PostulanteService, private router: Router) {
    
    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable()
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(postulanteFromService => {
      this.postulanteLocal = JSON.parse(JSON.stringify(postulanteFromService));
      if (this.postulanteLocal.rutacv != null && this.postulanteLocal.rutacv != "") {
        this.tieneArchivoCv = true
        // Obtener el nombre del archivo sin los caracteres especiales
        this.nombreCvSubido = decodeURIComponent(this.postulanteLocal.rutacv.substring(this.postulanteLocal.rutacv.lastIndexOf('/') + 1));
      } else {
        this.tieneArchivoCv = false
      }
    });
  }




  ngOnInit(): void {
  }

  // salir del flow de carga de CV
  volver(): void {
    this.router.navigate([ROUTES.RESUMEN]);
  }
  // ir a vista de input para seleccionar archivo de CV
  agregar(): void {
    this.tieneArchivoCv = false
  }
  /**
   * eliminar el archivo de CV subido 
   */
  async eliminar(): Promise<void> {
    try {
      this.postulanteService.deleteFileCv();
      this.tieneArchivoCv = false
      this.httpError = false;
    } catch (error) {
      console.error(error);
      this.httpError = true;
    }
  }

  /**
   * guardar archivo de CV seleccionado
   */
  async guardar(): Promise<void> {
    if (!this.selectedFile)
      return

    const formData = new FormData();
    formData.append('file', this.selectedFile, this.selectedFile.name);

    try {
      this.postulanteService.postFileCv(formData)
      this.httpError = false;
    } catch (error) {
      this.httpError = true;
    }
  }

  /**
   * Obtiene el File seleccionado en el input file
   * @param event eveto que se dispara cada vez que cambia la selección de archivo del select
   */
  selectFile(event: any): void {
    const fileInput = event.target;
    const file: File = fileInput.files[0];

    if (file) {
      this.selectedFile = file;
    } else {
      this.selectedFile = null;
    }
  }

  /**
   * Descarga del backend del archivo de CV subido 
   * @returns
   */
  async descargar(): Promise<void> {
    try {
      const cvFile = await this.postulanteService.getFileCv()
      if(cvFile == null)
        return
      const urlBlob = window.URL.createObjectURL(cvFile);
      const link = document.createElement('a');
      link.href = urlBlob;
      link.download = this.nombreCvSubido;
      link.click();
      window.URL.revokeObjectURL(urlBlob);
    } catch (error) {
      throw error
    }
  }
}
