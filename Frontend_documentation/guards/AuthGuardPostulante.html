import { Injectable } from '@angular/core';
import { CanActivate, ActivatedRouteSnapshot, RouterStateSnapshot, Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
import { PostulanteService } from '../servicios/postulante.service';

@Injectable({
  providedIn: 'root'
})
export class AuthGuardPostulante implements CanActivate {

  // DATOS POSTULANTE SERVICE
  existPostulante$: Observable<boolean>;
  existPostulanteSuscripcion: Subscription;
  notHaveProfile: boolean = true;

  constructor(private postulanteService: PostulanteService, private router: Router) {
    this.existPostulante$ = this.postulanteService.tienePerfil.asObservable();
    this.existPostulanteSuscripcion = this.existPostulante$.subscribe(datoTienePerfil => {
      this.notHaveProfile = !datoTienePerfil;
    })
  }

  /**
   * Define como regla que el usuario no tenga un perfil cargado, 
   * utilizando informacion albergada en el PostulanteService.
   * @param next 
   * @param state 
   * @returns boolean
   */
  canActivate(
    next: ActivatedRouteSnapshot,
    state: RouterStateSnapshot): boolean {
    if (this.notHaveProfile) { // Supongamos que tienes un método isLoggedIn en tu servicio de autenticación
      // Aquí puedes realizar cualquier otra lógica de restricción basada en tus datos de servicio
      return true; // El usuario tiene permiso para acceder a la ruta
    } else {
      // Redirigir al usuario a la página de inicio de sesión si no está autenticado
      this.router.navigate(['/']);
      return false; // El usuario no tiene permiso para acceder a la ruta
    }
  }
}
