import { Injectable } from '@angular/core';
import { EducacionPostulante } from '../interfaces/EducacionPostulante';
import { HttpClient } from '@angular/common/http';
import { API_BASE, API_PATH } from 'src/constants';
import { BehaviorSubject, Observable, Subscription } from 'rxjs';
import { PostulantePerfil } from '../interfaces/PostulantePerfil';
import { PostulanteService } from './postulante.service';
import { Curso } from '../interfaces/Curso';
import { ExperienciaLaboral } from '../interfaces/ExperienciaLaboral';
import { ConocimientoInformatica } from '../interfaces/ConocimientoInformatica';
import { DatosAdicionales } from '../interfaces/DatosAdicionales';
import { IdiomaPostulante } from '../interfaces/IdiomaPostulante';
import { EducacionPostulanteRequestDTO } from '../interfaces/EducacionPostulanteRequestDTO';

@Injectable({
  providedIn: 'root'
})
export class DatosAdicionalesService {

  /**
   * Declaracin e inicializacin de objetos que son atributos del perfil del postulante
   */
  private educacionPostulante: EducacionPostulante[] = [];
  private curso: Curso[] = [];
  private experienciaLaboral: ExperienciaLaboral[] = [];
  private conocimientosInformatica: ConocimientoInformatica[] = [];
  private idioma: IdiomaPostulante[] = [];

  /**
   * Declaracin e inicializacin de URLs para ruteo y navegacin
   */
  private urlPostDatosCV: string = `${API_BASE}/${API_PATH.DATOS_CV}`;
  private urlPutDatosCV: string = `${API_BASE}/${API_PATH.DATOS_CV}`;
  private urlGetDatosCVPorIdPostulante: string = `${API_BASE}/${API_PATH.DATOS_CV}`;

  /**
   * Variables de control del postulantePerfil
   */
  postulanteObservable$: Observable<PostulantePerfil>;
  perfilPostulanteSuscription: Subscription;
  postulanteForm!: PostulantePerfil;

  /**
   * Estado que mantiene los datos del cv de postulante del lado del cliente
   */
  private _datosCVLocal!: DatosAdicionales;
  datosCVLocal: BehaviorSubject<DatosAdicionales> = new BehaviorSubject<DatosAdicionales>(this._datosCVLocal);

  /**
   * Estado que notifica si el usuario tiene o no datos adicionales
   */
  private _noTieneDatosCV: boolean = true;
  noTieneDatosCV: BehaviorSubject<boolean> = new BehaviorSubject<boolean>(this._noTieneDatosCV)

  constructor(private http: HttpClient, private postulanteService: PostulanteService) {
    this.postulanteObservable$ = this.postulanteService.postulanteLocal.asObservable();
    this.perfilPostulanteSuscription = this.postulanteObservable$.subscribe(perfilFromService => {
      this.postulanteForm = { ...perfilFromService };
      this.blanqueoDeDatosCV();
      this.updateEstadoDatosCVLocal(this._datosCVLocal);
      this.obtenerDatosCVByIdPostulante();
    });
  }

  /**
   * Obtiene los DatosCV almacenado en el backend
   * Actualiza los estados tieneDatosCV y datosCV
   * @param id es un id de postulante que viene dado por el login
   */
  async obtenerDatosCVByIdPostulante() {
    try {
      //Checkeamos undefined porque el atributo idpostulante es opcional, y el NaN porque el idpostulante cuando se
      // inicializa, lo hace con valor NaN. 
      if (this.postulanteForm.idpostulante === undefined || Number.isNaN(this.postulanteForm.idpostulante)) {
        console.log("No es posible obtener los datos adiconales, ya que el idpostulante es: " + this.postulanteForm.idpostulante);
        return
      }
      const datosCVByPostulante: DatosAdicionales = await this.getDatosCVPorIdPostulante(this.postulanteForm.idpostulante);

      if (datosCVByPostulante.conocimientosInformatica.length === 0 &&
        datosCVByPostulante.educacionPostulante.length === 0 &&
        datosCVByPostulante.experienciaLaboral.length === 0 &&
        datosCVByPostulante.idioma.length === 0 &&
        datosCVByPostulante.curso.length === 0) {
        this.updateEstadoNoTieneDatosCV(true);
      } else {
        this.updateEstadoNoTieneDatosCV(false);
      }
      this.updateEstadoDatosCVLocal(datosCVByPostulante)

    } catch (error) {
      this.updateEstadoNoTieneDatosCV(true);
      console.error("Error al obtener los datos adicionales: ", error);
    }
  }

  /**
   * Se encarga de hacer el HTTP GET a la API del backend para traer los datos adicionales del usuario
   * Segun su idpostulante
   * @param id de usuario
   * @returns promesa con el perfil del postulante si es que lo encuentra
   */
  async getDatosCVPorIdPostulante(id: number): Promise<DatosAdicionales> {
    try {
      const response = await this.http.get<any>(`${this.urlGetDatosCVPorIdPostulante}/${id}`).toPromise();
      return response;
    } catch (error: any) {
      if (error.status === 404) {
        console.error(`El postulante con el id ${id}, no posee un datos adicionales`, error);
      } else {
        console.error('Error desconocido', error);
      }
      throw error;
    }
  }

  /**
 * Persiste contra la API los datos adicionales
 */
  async persistDatosCV() {
    try {
      const newItemResponse: DatosAdicionales = await this.postDatosAdicionales();
      // sincroniza el nuevo valor del estado de los Datos Adicionales y lo emite
      this.updateEstadoDatosCVLocal(newItemResponse)
    } catch (error) {
      // manejo de errores
      console.log('postulanteService, method persistPostulante. Error: ', error);
      throw (error)
    }
  }

  /**
 * Se encarga de hacer el HTTP POST a la API del Backend
 * @returns retorno de la promesa con el perfil del postulante
 */
  async postDatosAdicionales(): Promise<DatosAdicionales> {
    const educacionesDTO: EducacionPostulanteRequestDTO[] = this.convertirAEducacionPostulanteRequestDTO(this._datosCVLocal.educacionPostulante);
    let postData: any =
    {
      idpostulante: this._datosCVLocal.idpostulante,
      educacion: educacionesDTO,
      cursos: this._datosCVLocal.curso,
      expLaborales: this._datosCVLocal.experienciaLaboral,
      idiomas: this._datosCVLocal.idioma,
      informaticas: this._datosCVLocal.conocimientosInformatica
    }
    try {
      const response = await this.http.post<any>(this.urlPostDatosCV, postData).toPromise();
      return response;

    } catch (error) {
      console.error('No se pudo postear los datos adicionales del postulante ', error)
      throw error;
    }
  }
  /**
   * Actualizar datos de educacion postulante
  */
  async actualizarEducacionPostulante() {
    try {
      // postea al backend y obtiene la respuesta
      let postData = [...this._datosCVLocal.educacionPostulante];
      let urlUpdate = `${this.urlPutDatosCV}/educacion/${this._datosCVLocal.idpostulante}`
      try {
        const newEducacionPostulante = await this.http.put<any>(urlUpdate, postData).toPromise();
        this._datosCVLocal.educacionPostulante = newEducacionPostulante;
        // sincroniza el nuevo valor del estado del perfil del postulante y lo emitirlo
        this.updateEstadoDatosCVLocal(this._datosCVLocal)
      } catch (error) {
        console.error('No se pudo actualizar los datos adicionales ', error)
        throw error;
      }

    } catch (error) {
      // manejo de errores
      console.log('datosAdicionalesService, method update. Error: ', error);
      throw (error)
    }
  }

  /**
  * Actualizar datos de Curso postulante
  */
  async actualizarCursoPostulante() {

  async actualizarExpLaboralPostulante() {

  /**
* Actualizar datos de Idioma postulante
*/
  async actualizarIdiomaPostulante() {
    try {

  /**

  /**
   * Se encargara de mapear el arreglo local de EducacionPostulante a la clase 
   * EducacionPostulanteRequestDTO del backend.
   * @param arreglo EducacionPostulante[]
   * @returns EducacionPostulanteRequestDTO[]
   */
  private convertirAEducacionPostulanteRequestDTO(arreglo: EducacionPostulante[]): EducacionPostulanteRequestDTO[] {
    const arregloDTO: EducacionPostulanteRequestDTO[] = [];
    for (const educacion of arreglo) {
      const educacionDTO: EducacionPostulanteRequestDTO = {
        idpostulante: this._datosCVLocal.idpostulante,
        idniveleducativo: educacion.idniveleducativo.idniveleducativo,
        especialidad: educacion.especialidad,
        establecimiento: educacion.establecimiento,
        estado: educacion.estado
      };
      arregloDTO.push(educacionDTO);
    }
    return arregloDTO;
  }

  // ************************************* ACUTALIZACION DE ESTADOS ***********************************************/

  /**
   * Actualiza el estado interno de los datos adicionales del postulante
   * con un nuevo valor. El cual se emite a todos los suscriptores
   * @param newDatosCV nuevos valores de los datos adicionales
   */
  updateEstadoDatosCVLocal(newDatosCV: DatosAdicionales): void {
    this._datosCVLocal = newDatosCV;
    this.datosCVLocal.next(this._datosCVLocal);
    if (this._datosCVLocal.conocimientosInformatica.length === 0 &&
      this._datosCVLocal.educacionPostulante.length === 0 &&
      this._datosCVLocal.experienciaLaboral.length === 0 &&
      this._datosCVLocal.idioma.length === 0 &&
      this._datosCVLocal.curso.length === 0) {
      this.updateEstadoNoTieneDatosCV(true);
    } else {
      this.updateEstadoNoTieneDatosCV(false);
    }
  }

  /**
   * Actualiza el estado interno de tienePerfil y lo emite a los suscriptores
   * @param newState nuevo valor boolean
   */
  updateEstadoNoTieneDatosCV(newState: boolean): void {
    this._noTieneDatosCV = newState;
    this.noTieneDatosCV.next(this._noTieneDatosCV)
  }

  /**
   * Blanquea los Datos de Adiconales 
   */
  blanqueoDeDatosCV() {
    if (this.postulanteForm.idpostulante != undefined) {
      this._datosCVLocal = {
        idpostulante: this.postulanteForm.idpostulante,
        educacionPostulante: this.educacionPostulante,
        experienciaLaboral: this.experienciaLaboral,
        conocimientosInformatica: this.conocimientosInformatica,
        curso: this.curso,
        idioma: this.idioma,
      }
    } else {
      console.log("Error. No se encuentra el ID del Postulante")
    }

  }

}
