import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { API_BASE, API_PATH, MAXIMOS } from 'src/constants';

import { Localidad } from '../interfaces/Localidad';
import { Calle } from '../interfaces/Calle';

import { Pais } from '../interfaces/Pais';

import { Idioma } from '../interfaces/Idioma';

    calle: `${API_BASE}/${API_PATH.CALLE}`,
    genero: `${API_BASE}/${API_PATH.GENERO}`,
    pais: `${API_BASE}/${API_PATH.NACIONALIDAD}`,
    tipo_doc: `${API_BASE}/${API_PATH.TIPO_DNI}`,

    idioma: `${API_BASE}/${API_PATH.IDIOMA}`
  }
  filterTerms: any = ['desconocido', 'desconocida']

  private cache: { [resource: string]: any } = {}; // Objeto para almacenar los datos en caché

  constructor(private http: HttpClient) { }

  async getData<T>(path: string): Promise<T> {
    // Verificar si los datos ya están en caché
    if (this.cache[path]) {
      return this.cache[path];
    }
    // si no esta cacheado hace el fetch
    let intento = 0;
    while (intento < MAXIMOS.MAX_INTENTOS_FETCH) {
      try {
        const response = await this.http.get<T>(path).toPromise();
        if (response) {
          this.cache[path] = response;
          return response;
        } else {
          throw new Error('Respuesta indefinida');
        }
      } catch (error: any) {
        console.error(`Error en el intento ${intento + 1} de ${MAXIMOS.MAX_INTENTOS_FETCH}: ${error}`);
        intento++;
      }
    }
    throw new Error(`No se pudo obtener la respuesta después de ${MAXIMOS.MAX_INTENTOS_FETCH} intentos`);
  }

  async getProvincias(): Promise<Provincia[]> {
    let url: string = this.urlOption.provincia
    try {
      const provincias: Provincia[] = await this.getData<Provincia[]>(url);
      // const fitredData: Provincia[] = this.filterOptions(this.filterTerms.provincia, provincias)
      // return fitredData;
      return provincias;
    } catch (error) {
      console.error('Error al obtener datos de ' + url, error);
      throw error;
    }
  }

  //*********************************** FILTER FUNCTION ************************************/

  filterOptions(terms: string[], items: any[]): any[] {

    let newArray = [...items]
    terms.forEach((term) => {
      newArray = newArray.filter((item) => {
        return item.descripcion.toLowerCase() != term.toLowerCase()
      })
    })
    return newArray;
  }
