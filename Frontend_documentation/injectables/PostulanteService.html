import { Injectable } from '@angular/core';
import { TipoDocumento } from '../interfaces/TipoDocumento';
import { Genero } from '../interfaces/Genero';
import { Pais } from '../interfaces/Pais';
import { Partido } from '../interfaces/Partido';
import { Localidad } from '../interfaces/Localidad';
import { Calle } from '../interfaces/Calle';
import { Provincia } from '../interfaces/Provincia';
import { EstadoCivil } from '../interfaces/EstadoCivil';
import { BehaviorSubject } from 'rxjs';
import { HttpClient } from '@angular/common/http';
import { API_BASE, API_PATH } from 'src/constants';
import { PostulantePerfil } from '../interfaces/PostulantePerfil';
import { FileData } from '../interfaces/FileData';

@Injectable({
  providedIn: 'root'
})

/**
 * PostulanteService
 * Tiene las sigueintes responsabilidades:
 *  1. Mantener estado interno de perfil de usuario, Ofrecer y emitir el valor actualizado a los sucriptores.
 *  2. ABM de perfil de usuario contra la API.
 *  3. Estado interno booleano para indica si el idUsuario actual tiene o no perfil de usuario en el backend. El estado se ofrece como observable.
 *  4. GET, POST y DELETE de archvivo de CV. Llevar estado interno de archivo de CV y notificarlo a suscriptores
 */
export class PostulanteService {
  /**
   * Declaración e inicialización de objetos que son atributos del perfil del postulante
   */
  private tipoDni: TipoDocumento = { idTipodoc: NaN, descripcion: '' };
  private tipoGenero: Genero = { idGenero: NaN, descripcion: '' };
  private estadoCivil: EstadoCivil = { idEstadoCivil: NaN, descripcion: '', };
  private pais: Pais = { idPais: NaN, descripcion: '' };
  private provincia: Provincia = { idProvincia: NaN, descripcion: '' };
  private partido: Partido = { idpartido: NaN, descripcion: '' };
  private localidad: Localidad = { idLocalidad: NaN, codPostal: '', descripcion: '' };
  private calle: Calle = { idCalle: NaN, descripcion: '' };
  private foto: FileData = { name: '', extension: '', data: '', path: '' }
  /**
   * Declaración e inicialización de URLs para ruteo y navegación
   */
  private urlPostPostulante: string = `${API_BASE}/${API_PATH.POSTULANTE}`
  private urlgetPostulantePorIdUsuario: string = `${API_BASE}/${API_PATH.POSTULANTE_POR_USUARIO}`
  /**
   * Estado que mantiene el id de usuario.
   * En sircunstancias normales se obtiene con el Login
   */
  private idUsuario: number = 6;
  /**
   * Estado que mantiene los datos del perfil de postulante del lado del cliente
   */
  private _postulanteLocal!: PostulantePerfil;
  postulanteLocal: BehaviorSubject<PostulantePerfil> = new BehaviorSubject<PostulantePerfil>(this._postulanteLocal);
  /**
   * Estado que notifica si el usuario tiene o no perfil
   */
  private _tienePerfil: boolean = false;
  tienePerfil: BehaviorSubject<boolean> = new BehaviorSubject<boolean>(this._tienePerfil)





  constructor(private http: HttpClient) {
    this.blanqueoDePostulanteLocal()
    // ACTUALIZA POR PRIMERA VEZ A TODOS LOS SUBCRIPTOS CON UN POSTULANTE VACIO
    this.updateEstadoPostulanteLocal(this._postulanteLocal);
    // OBTENEMOS EL POSTULANTE REAL
    this.obtenerPostulanteByIdUsuario(this.idUsuario)
  }







  // ************************************* HTTP REQUEST Functions ***********************************************/

  /**
   * Obtiene el perfil de un postulante almacenado en el backend
   * Actualiza los estados tienePerfil y postulantePerfil
   * @param id es un id de usuario que viene dado por el login
   */
  async obtenerPostulanteByIdUsuario(id: number) {
    try {
      this.idUsuario = id;
      const perfilByUsuario: PostulantePerfil = await this.getPostulantePorIdUsuario(id);

      this.updateEstadoTienePerfil(true)
      this.updateEstadoPostulanteLocal(perfilByUsuario)

      console.log(perfilByUsuario);

    } catch (error) {
      this.updateEstadoTienePerfil(false)

      this.blanqueoDePostulanteLocal();
      this.postulanteLocal.next(this._postulanteLocal);

      console.error(error);
    }
  }

  /**
   * Se encarga de hacer el HTTP GET a la API del backend para traer un perfil de postulante con un id de usuario
   * @param id de usuario
   * @returns promesa con el perfil del postulante si es que lo encuentra
   */
  async getPostulantePorIdUsuario(id: number): Promise<PostulantePerfil> {
    try {
      const response = await this.http.get<any>(`${this.urlgetPostulantePorIdUsuario}/${id}`).toPromise();
      return response;
    } catch (error: any) {
      if (error.status === 404) {
        console.error(`El postulante con el id de Usuario ${id}, no posee un perfil cargado`, error);
      } else {
        console.error('Error desconocido', error);
      }
      throw error;
    }
  }

  /**
   * Persiste contra la API un nuevo perfil de postulante
   * Actualia los estados internos de postulantePerfil y tienePerfil
   */
  async persistPostulante() {
    console.log(this._postulanteLocal);
    try {
      // 1. modificar la data de la foto de perfil si es que tiene
      const dataFotoBase64 = this._postulanteLocal.foto.data;
      if (dataFotoBase64 != "") {
        this._postulanteLocal.foto.data = dataFotoBase64.split(',')[1];
      }
      // 2. postea al backend y obtiene la respuesta
      const newPostulanteResponse: PostulantePerfil = await this.postPostulante();
      // 3. sincroniza el nuevo valor del estado del perfil del postulante y lo emitirlo
      this.updateEstadoPostulanteLocal(newPostulanteResponse)
      // 4. sincroniza el nuevo estado de tienePerfil a true, y lo emitirlo
      this.updateEstadoTienePerfil(true)

    } catch (error) {
      // 4. manejo de errores
      console.log('postulanteService, method persistPostulante. Error: ', error);
      throw (error)
    }
  }

  /**
   * Se encarga de hacer el HTTP POST a la API del Backend
   * @returns retorno de la promesa con el perfil del postulante
   */
  async postPostulante(): Promise<PostulantePerfil> {
    let postData = { ... this._postulanteLocal }

    try {
      const response = await this.http.post<any>(this.urlPostPostulante, postData).toPromise();
      return response;

    } catch (error) {
      console.error('No se pudo postear el nuevo postulante ', error)
      throw error;
    }
  }

  /**
   * Actualizar datos de postulante
   */
  async actualizarPerfilPostulante() {
    console.log(this._postulanteLocal);
    try {
      // 1. modificar la data de la foto de perfil si es que tiene
      const dataFotoBase64 = this._postulanteLocal.foto.data;
      if (dataFotoBase64 != null && dataFotoBase64 != "") {
        this._postulanteLocal.foto.data = dataFotoBase64.split(',')[1];
      }
      // 2. postea al backend y obtiene la respuesta
      let postData = { ... this._postulanteLocal }
      let urlUpdate = `${this.urlPostPostulante}/${this._postulanteLocal.idpostulante}`
      try {
        const newPostulanteResponse = await this.http.put<any>(urlUpdate, postData).toPromise();
        // 3. sincroniza el nuevo valor del estado del perfil del postulante y lo emitirlo
        this.updateEstadoPostulanteLocal(newPostulanteResponse)
      } catch (error) {
        console.error('No se pudo actualizar el perfil del postulante ', error)
        throw error;
      }

    } catch (error) {
      // 4. manejo de errores
      console.log('postulanteService, method persistPostulante. Error: ', error);
      throw (error)
    }
  }

  // ************************************* ACUTALIZACION DE ESTADOS ***********************************************/

  /**
   * Actualiza el estado interno del perfil del postulante con un nuevo valor
   * Emite el nuevo valor a todos los suscriptores
   * @param newPostulante nuevos valores del perfil postulante
   */
  updateEstadoPostulanteLocal(newPostulante: PostulantePerfil): void {
    this._postulanteLocal = newPostulante;
    this.postulanteLocal.next(this._postulanteLocal)
  }

  /**
   * actualiza el estado interno de tienePerfil y lo emite a los suscriptores
   * @param newState nuevo valor boolean
   */
  updateEstadoTienePerfil(newState: boolean): void {
    this._tienePerfil = newState;
    this.tienePerfil.next(this._tienePerfil)
  }

  // ************************************* ADMINISTRAR CV FILE ***********************************************/

  /**
   * Descarga archivo de CV desde el backend
   * @returns 
   */
  async getFileCv(): Promise<Blob | null> {
    try {
      const response = await this.http.get(`${API_BASE}${this._postulanteLocal.rutacv}`, { responseType: 'blob' }).toPromise();
      if (response) {
        const cvFile = new Blob([response], { type: 'application/pdf' });
        return cvFile
      } else {
        console.error('La respuesta de la solicitud HTTP es undefined.');
        return null
      }
    } catch (error) {
      console.error(error)
      throw error;
    }
  }

  /**
   * Acción HTTP Request DELETE de archivo de CV subido
   */
  async deleteFileCv(): Promise<void> {
    this.http.delete<any>(`${API_BASE}/${API_PATH.SUBIR_CV}/${this._postulanteLocal.idpostulante}`).subscribe(
      () => {
        this.obtenerPostulanteByIdUsuario(this._postulanteLocal.idusuario)
      },
      (error) => {
        throw error;
      }
    );
  }

  /**
   * Acción HTTP Request POST para subir archivo al backend
   */
  async postFileCv(formData: FormData): Promise<void> {
    this.http.post<any>(`${API_BASE}/${API_PATH.SUBIR_CV}/${this._postulanteLocal.idpostulante}`, formData).subscribe(
      (response) => {
        this.obtenerPostulanteByIdUsuario(this._postulanteLocal.idusuario)
      },
      (error) => {
        throw error;
      }
    );
  }

  // ************************************* FUNCIONES AUXILIARES ***********************************************/

  /**
   * Blanquea los valores del perfil del postulante
   */
  blanqueoDePostulanteLocal(): void {
    this._postulanteLocal = {
      idusuario: this.idUsuario,
      idpostulante: NaN,
      fechaalta: '',
      rutacv: '',
      estado: false,
      nombres: '',
      apellido: '',
      nombreautopercibido: '',
      fecnac: '',
      tipoDoc: this.tipoDni,
      nrodoc: NaN,
      nrofiscal: NaN,
      genero: this.tipoGenero,
      otro_genero: '',
      pais: this.pais,
      estadoCivil: this.estadoCivil,
      foto: this.foto,
      tienelicencia: false,
      tienehijos: false,
      telefono: NaN,
      celular: NaN,
      mail: '',
      partido: this.partido,
      provincia: this.provincia,
      localidad: this.localidad,
      calle: this.calle,
      numcalle: NaN,
      areasIncumbencia: [],
      filesize: NaN
    };
  }
}
